\chapter{Cotorsion pairs and chain complexes}

In the previous chapter we have seen the various interaction between cotorsion pairs and model structures on Grothendieck categories, we will now see how we can use this knowledge to study chain complexes in these categories. The theory has been deeveloped by James Gillespie in \cite{Gillespie2004Flat} and \cite{Gillespie2006Quillen} and by Xiaoyan Yang and Nanqing Ding in \cite{DingYang2014}.


\section{Initial remarks}

We begin by setting a standard notation for this chapter.

\begin{notation}
    Let $\C$ be an abelian category, let $A \in \ob(\C)$ and let $n \in \Z$. We denote as $S^n A$ the object of $\Ch(\C)$ given by the complex whose only nonzero element is $A$ in position $n$; we also denote as $D^n A$ the object of $\Ch(\C)$ given by the complex whose only nonzero elements are $A$ in position $n-1, n$  and whose only nonzero morphsim is $d_{n-1} = \id_A$.\\ 
    If $X = (X_n, d_n)$ is a chain complex in $\Ch(\C)$, we denote as $Z_nX$ the object in $\C$ given by $\ker d_n$, while we denote as $B_nX$ the object in $\C$ given by $\im d_{n+1}$; we also denote $Z_nX/B_nX$, the $n$-th homology of the complex, as $H_n X$.
    Finally, letting $Y$ be another complex, we denote as $\Hom(X, Y)$ the complex 
    \[\begin{tikzcd}
        \cdots & {\prod_{k \in \Z}\Hom_\C(X_k, Y_{k+n})} & {\prod_{k \in \Z}\Hom_\C(X_k, Y_{k+n-1})} & {\cdots,}
        \arrow[from=1-1, to=1-2]
        \arrow["{\delta_n}", from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
    \end{tikzcd}\]
    where, given $f \in {\prod_{k \in \Z}\Hom_\C(X_k, Y_{k+n})}$, we set $(\delta_n f)_k =  d^Y_{n+k} f_k - (-1)^n f_{k-1}d^X_k$.  \\
    Lastly, we define the complex $\Sigma^n X$ as the complex whose objects are $(\Sigma^nX)_k = X_{n-k}$ and whose differential is $d^{\Sigma^n X}_k = (-1)^n d^X_{n-k}$.
\end{notation}



We continue by proving some standard results about categories of chain complexes.


\begin{definition}[Degreewise Ext]
    Let $\C$ be an abelian category, let $X,Y$ be objects in $\Ch(\C)$, we define the \textbf{degreewise Ext group} $\Ext^1_{dw}(X, Y)$ as the subgroup of $\Ext^1_{\Ch(\C)}(X, Y)$ given by all extensions which are split degreewise. 
\end{definition}

\begin{lemma}
    \label{lem:deg-ext}
    Let $\C$ be an abelian category, let $X,Y$ be objects in $\Ch(\C)$. We have that
    \[\Ext^1_{dw}(X, \Sigma^{-n-1} Y) \cong H_n\Hom(X,Y) \cong  \Hom_{\Ch(\C)}(X, \Sigma^{-n} Y) / \sim,\]
    where $\sim$ denotes chain homotopy.
\end{lemma}

\begin{proof}
    Let $f=(f_k)_k$ be an element of $\Hom(X,Y)_n$. We see that $f \in \ker \delta_n$ if and only if $d^Y_{n+k} f_k = (-1)^n f_{k-1}d^X_k $, but $(-1)^n d^Y_{n+k}$ is exactly $d^{\Sigma^{-n} Y}_k$, therefore the condition is equivalent to checking if $f$ is a morphism between $X$ and $\Sigma^{-n} Y$. Similarly, the elements of $\im \delta_{n+1}$ are of the form 
    \[d^Y_{k+n+1}f_k - (-1)^{n+1}f_{k-1}d^X_k = (-1)^n d^{\Sigma^{-n} Y}_{k+1}f_k + (-1)^n f_{k-1}d^X_k,\]
    thus they are exactly null-homotopic maps from $X$ to $\Sigma^{-n}Y$; therefore $H_n\Hom(X,Y) \cong  \Hom_{\Ch(\C)}(X, \Sigma^{-n} Y) / \sim$. \\
    We procced by consideering the complex $W = \Sigma^{-n-1} Y$ and an extension
    \[\begin{tikzcd}
        0 & W & Z & X & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    that is split degreewise. This implies that $Z_k \cong W_k \oplus X_k$, therefore by the commutativity of the diagram
    \[\begin{tikzcd}
        {W_k} & {W_k\oplus X_k} & {X_k} \\
        {W_{k-1}} & {W_{k-1} \oplus X_{k-1}} & {X_{k-1}}
        \arrow[from=1-1, to=1-2]
        \arrow["{d^W_k}", from=1-1, to=2-1]
        \arrow[from=1-2, to=1-3]
        \arrow["{d^Z_k}", from=1-2, to=2-2]
        \arrow["{d^X_k}", from=1-3, to=2-3]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
    \end{tikzcd}\]
    we know that 
    \[d_k^Z =  \begin{bmatrix}
        d_k^W & \phi_k \\
        0 & d^k_X 
        \end{bmatrix}  \]
    for some $\phi_k \colon X_k \longrightarrow W_{k-1} = \Sigma^{-n} Y$; thus we have found an element $\phi = (\phi_k)_k \in \prod_{k} \Hom(X_k, Y_{n+k})$. If we substitute the form of $d^W_k$ we just derived in the condition $d^W \circ d^W = 0$ we get that $d^Y_{k+n} \phi_k - (-1)^n \phi_k d_k^X = 0$, i.e. $\phi \in Z_k \Hom(X,Y)$.\\
    Similarly, two extensions $Z, Z'$ are equivalent iff there exists a chain isomorphism $h$ such that
    \[\begin{tikzcd}
        {W} & {Z} & {X} \\
        {W} & {Z'} & {X}
        \arrow[from=1-1, to=1-2]
        \arrow["{\id_W}", from=1-1, to=2-1]
        \arrow[from=1-2, to=1-3]
        \arrow["{h}", from=1-2, to=2-2]
        \arrow["{\id_X}", from=1-3, to=2-3]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
    \end{tikzcd}\]
    commutes, hence
    \[h_k =  \begin{bmatrix}
        1 & \psi_k \\
        0 & 1 
        \end{bmatrix}  \]
    for all $k$. If we denote as $\phi, \phi'$ the elements of $Z_k \Hom(X,Y)$ induced by $Z, Z'$, some quick matrix calculations show that $h$ is actually a morphism of chain complexes iff
    \[\phi'_k - \phi = (-1)^{n+1}d^Y_{k+n+1} h_k -h_{k-1} d^X_k.\]
    We conclude by remarking that, up to sign, this is equivalent to asking that $\phi - \phi' \in  B_k \Hom(X,Y)$.

    
\end{proof}


\begin{lemma}
    \label{lem:chain-adjoints}
    Let $\C$ be an abelian category, let $A \in ob(\C)$ and let $X,Y \in \ob(\Ch(\C))$. The following natural isomorphisms hold:
    \begin{enumerate}
		\itemsep0em 
		\item $\HomCh{\C}(D^n A, Y) = \Hom_\C(A, Y_n)$.
		\item $\HomCh{\C}(X, D^n A) = \Hom_\C(X_{n-1}, A)$.
		\item $\HomCh{\C}(S^n A, Y) = \Hom_\C(A, Z_nY)$.
		\item $\HomCh{\C}(X, S^n A) = \Hom_\C(X/B_nX, A)$.
		\item $\Ext^1_{\Ch(\C)}(D^n A, Y) = \Ext^1_\C(A, Y_n)$
		\item $\Ext^1_{\Ch(\C)}(X, D^{n+1} A) = \Ext^1_\C(X_n, A)$ 
		\item $\Ext^1_{\Ch(\C)}(S^n A, Y) = \Ext^1_\C(A, Z_nY) $
		\item $\Ext^1_{\Ch(\C)}(X, S^n A) = \Ext^1_\C(Z_nX, A)$ 
	\end{enumerate}
\end{lemma}





\begin{lemma}
    If $\C$ is a Grothendieck category, then $\Ch(\C)$ is also a Grothendieck category.
\end{lemma}


\begin{proof}
    Let $G$ be a generator of $\C$, the adjunction \[\HomCh{\C}(D^n G, X) = \Hom_\C(G, X_n)\] shows that $\{D^n G\}_{n \in \Z}$ is a generating family for $\Ch(\C)$. We conclude by remembering that colimits in $\Ch(\C)$ are taken dimensionwise.
\end{proof}




\section{Induced cotorsion pairs}
Let $\C$ be an abelian category, let $(\D, \E)$ be a cotorsion pair.

\begin{definition}[Induced cotorsion pair]
    Let $X$ be a chain complex.
    \begin{enumerate}
		\itemsep0em 
		\item $X$ is called a $\D$ \textbf{complex} if it is exact and $Z_n X \in \D$ for all $n \in \Z$.
		\item $X$ is called a $\E$ \textbf{complex} if it is exact and $Z_n X \in \E$ for all $n \in \Z$.
		\item $X$ is called a dg-$\D$ \textbf{complex} if $X_n \in \D$ for all $n \in \Z$ and $\Hom(X, B)$ is exact for all $\E$ complexes $B$.
		\item $X$ is called a dg-$\E$\textbf{complex} if $X_n \in \E$ for all $n \in \Z$ and $\Hom(A, X)$ is exact for all $\D$ complexes $A$.
	\end{enumerate}
    We denote the class of all $\D$ complexes as $\widetilde{\D}$, while the class of all dg-$\D$ complexes is denoted as $\dg{\D}$. Similarly, we denote the class of all $\E$ complexes as $\widetilde{\E}$, while the class of all dg-$\E$ complexes is denoted as $\dg{\E}$. We also use $\ex$ to denote the class of all exact complexes.
\end{definition}

We begin with a fundamental result on this kinds of complexes.

\begin{proposition}
	\label{prop:dg-containment}
	Every morphism of complexes from a $\D$-complex to an $\E$-complex is null-homotopic.
\end{proposition}
\begin{proof}
	Let us consider $D\in \D$ and $E \in \E$ and chose a morphism $f\colon D \longrightarrow E$. We will show that there exists an homotopically equivalent morphism $g$ such that $g_{n} d_{n+1}^D = 0$ and $d_{n}^E g_n = 0$ for all $n \in \N$, where the subscript indicates the degree of the component; we will also denote with $Z_n f$ the induced morphism between $Z_n D$ and $Z_n E$. We begin by considering the exact sequence (given us by the exactness of $E$ at homological position $n+1$)
	\[\begin{tikzcd}
		0 & {Z_{n+1 }E} & {E_{n+1}} & {Z_n E} & {0.}
		\arrow[from=1-1, to=1-2]
		\arrow[from=1-2, to=1-3]
		\arrow["{d_{n+1}^E}", from=1-3, to=1-4]
		\arrow[from=1-4, to=1-5]
	\end{tikzcd}\]
	Applying the functor $\Hom_\C(Z_n D, {-})$ to such a sequence yields
	\[\begin{tikzcd}
		{\Hom_\C(Z_n D, E_{n+1})} & {\Hom_\C(Z_n D, Z_n E)} & {\Ext^1_\C(Z_n D, Z_{n+1 }E),}
		\arrow["{(d_{n+1}^E)_*}", from=1-1, to=1-2]
		\arrow[from=1-2, to=1-3]
	\end{tikzcd}\]
	therefore, by observing that $Z_n D \in \D$ and $Z_{n+1 }E \in \E$ implies the equality ${\Ext^1_\C(Z_n D, Z_{n+1 }E)} =0 $, we know that ${(d_{n+1}^E)_*}$ must be surjective; thus there exists a morphism $\alpha_n \colon Z_n D \longrightarrow E_{n+1}$ that satisfies $Z_n f = {(d_{n+1}^E)_*} \alpha_n$. With a similar argument (the vanishing of $\Ext$) we prove that the exactness of $D$ at position $n$, in the form of the exact sequence
	\[\begin{tikzcd}
		0 & {Z_{n }D} & {D_{n}} & {Z_{n-1} D} & {0,}
		\arrow[from=1-1, to=1-2]
		\arrow[from=1-2, to=1-3]
		\arrow["{d_{n}^D}", from=1-3, to=1-4]
		\arrow[from=1-4, to=1-5]
	\end{tikzcd}\]
	implies the surjectivity of the map $\Hom_\C(D_n,E_{n+1}) \longrightarrow \Hom_\C(Z_n D,E_{n+1}) $, therefore there exists $\beta_n \colon D_n \longrightarrow E_{n+1}$ such that $\alpha_n$ is the restriction of $\beta_n$ to $Z_n D$. We can thus define the morphism $g$ of complexes (that will be homotopically equivalent to $f$ by definition) component by component, as
	\[g_n = f_n + (d_{n+1}^E \beta_n - \beta_{n-1}d_n^D).\]
	By direct computation 
	\begin{align*}
		d_n^E g_n &= d_n^E f_n +d_n^E d_{n+1}^E \beta_n - d_n^E \beta_{n-1} d_n^D = \\
		& = d_n^E f_n - d_n^E \beta_{n-1} d_n^D = d_n^E f_n - f_{n-1}d_n^D =  0
	\end{align*} 
	and similarly $g_n d_{n+1}^D =0 $. We will now prove that this implies that $g$ is null-homotopic (and hence $f$ also is). We observe that $d_n^E g_n = 0$ implies that $\im(g_n) \subseteq Z_n E$ and $g_n d_{n+1}^D =0 $ implies that $g_n$ is also well defined up to quotienting with $B_n D = Z_n D$, therefore $g_n$ induces a morphism $\bar{g_n} D_n / Z_n D \longrightarrow Z_n E$, which makes the following diagram commute (the isomorphism is the first isomorphism theorem).
	\[\begin{tikzcd}
		{D_n} & {D_n} & {D_n} \\
		{B_{n-1}D = Z_{n-1}D} & {D_n/Z_nD} & {Z_n E}
		\arrow["{\id_{D_n}}", from=1-1, to=1-2]
		\arrow["{d_n^D}", from=1-1, to=2-1]
		\arrow["{\id_{D_n}}", from=1-2, to=1-3]
		\arrow[from=1-2, to=2-2]
		\arrow["{g_n}", from=1-3, to=2-3]
		\arrow["{\bar{d_n^D}}"', "\cong", from=2-2, to=2-1]
		\arrow["{\bar{g_n}}", from=2-2, to=2-3]
	\end{tikzcd}\]
	If we set $h_n = {\bar{g_n}}\left({\bar{d_n^D}}\right)^{-1} \colon Z_{n-1}D \longrightarrow Z_n E$, the argument used to define $\alpha$ shows that there exists $k_n \colon Z_{n-1} D \longrightarrow E_{n+1}$ such that $h_n = \left(d_{n+1}^E\right)_*k_n$. A quick calculation shows that $g = d_{n+1}^Ek_n d_n^D  - k_{n-1}d_{n-1}^D d_n^D$ , therefore $k_n d_n^D$ is the homotopy between $g$ and $0$ that we are looking for.
\end{proof}

It is important to remark that if $X \in \DD$ then it must be exact, so $X_n$ is an extension of $Z_nX$ and $Z_{n-1}X$; thus by lemma \ref{lem:cotorsion-trans}, $X_n \in \D$ for all $n \in \Z$. Similarly, if $X \in \EE$ then $X_n \in \E$ for all $n \in \Z$. This result, together with proposition \ref{prop:dg-containment}, implies that $\DD \subseteq \dg{\D}$ and $\EE \subseteq \dg{\E}$.

\begin{lemma}
    \label{lem:dg-sphere}
    Bounded below complexes (i.e. complexes $X$ where there exists $n\in \Z$ such that if $k<n$ then $X_k = 0$) with entries in $\D$ are dg-$\D$ complexes.\\
     Bounded above complexes (i.e. complexes $X$ where there exists $n\in \Z$ such that if $k>n$ then $X_k = 0$) with entries in $\E$ are dg-$\E$ complexes.
\end{lemma}

\begin{proof}
    We will prove the first statement. Let $X$ be a bounded below complex with entries in $\D$, we need to show that $\Hom(X, B)$ is exact for any $\E$ complex $B$. Given that if $B$ is an $\E$-complex then $\Sigma^k B$ is also an $\E$-complex, we only need to prove that all morphisms $f\colon X \longrightarrow B$ are null-homotopic. Let us assume without loss of generality that $X_n$ is the zero object for all negative $n$. We want to build a family of morphisms $h_k \colon X_k \longrightarrow E_{k+1}$ such that $f_k =d_{k+1}^E h_k+ h_{k-1}d^X_k$ for $k \in \Z$, as pictured below.

        \arrow["{{{h_0}}}"{description}, from=1-4, to=2-3]

    \[\begin{tikzcd}

        {E_{k+1}} & {E_k.}
        \arrow["{h_k}"', dashed, from=1-2, to=2-1]
        \arrow["{g_k}", from=1-2, to=2-2]

\end{proof}


\begin{proposition}
    If $(\D, \E)$ has enough $\D$-objects and enough $\E$-objects (in particular when it is complete), then $(\dg{\D}, \EE)$ and $(\DD, \dg{\E})$ are cotorsion pairs. 
\end{proposition}

\begin{proof}
    We will only prove the first statement; we begin by showing that $\rperp{(\dg{\D})} \subseteq \EE$. Let $X \in \rperp{(\dg{\D})}$, by applying lemmas \ref{lem:chain-adjoints} and \ref{lem:dg-sphere}, we get that, for every $D \in \D$,
    \[\Ext^1_\C(D, Z_n X) = \Ext^1_{\Ch(\C)}(S^n D, X) = 0,\]
    thus we have proven that $Z_n X \in \lperp{\D} =\E$ for all $n$. Let us now consider an epimorphism $f \colon D \longrightarrow Z_n X$ for some $n \in \Z$ and some $D \in \D$ (it must exists given that $(\D, \E)$ has enough $\D$-objects); as before $\Ext^1_{\Ch(\C)}(S^n D, X) = 0$, therefore the induced morphism $f'\colon S^n D \longrightarrow X$ must be null-homotopic and hence $f' = d^X_{n+1} h_n$ for some homotopy $\{h_i \colon (S^n D)_i \longrightarrow X_{i+1}\}_i$. This result implies that $Z_nX = \im(f') \subseteq \im(d^X_{n+1}) = B_n X$. We have concluded that $X$ must be eexact, thus $X \in \EE$. \\
    We continue by showing that $\lperp{\EE} \subseteq (\dg{\D})$. Let $X \in \lperp{\EE}$ and let $Y$ be a $\E$-complex, then $\Ext^1_{\Ch(\C)}(X,Y) = 0$ and in particular $\Ext^1_{dw}(X,Y) = 0$; so $\Hom(X,Y)$ is exact. To show that $X_n \in \D$ we can apply lemma \ref{lem:chain-adjoints} to write, for all $E\in \E$,
    \[\Ext^1_\C(X_n, E) \cong \Ext^1_{\Ch(\C)}(X,D^{n+1}E) = 0.\]
    The opposite inclusions are immediate consequences of lemma \ref{lem:deg-ext}
\end{proof}



\begin{definition}[Induced cotorsion pairs]
    Whenever the pairs $(\dg{\D}, \EE)$ and $(\DD, \dg{\E})$ are cotorsion pairs, they are said to be the \textbf{induced cotorsion pairs} (from $(\D, \E)$). We say that the induced cotorsion pairs are compatible if $\DD = \dg{\D} \cap \ex$ and $\EE = \dg{\E} \cap \ex$.
\end{definition}






\section{Hereditary cotorsion pairs}

To reach our objective of obtaining a model structure by applying Hovey's correspondence, we need the induced cotorsion pairs to be compatible and complete. The work of Xiaoyan Yang and Nanqing Ding (see \cite{DingYang2014}) has found a complete characterization of which cotorsion pairs in certain complete and bicomplete abelian category induce compatible and complete cotrsion pairs in the category of complexes.  We must remark that the paper originally claimed to settle the question in all complete and cocomplete abelian categories, as shown in \cite{herbera2025poisonousexample} by Dolors Herbera, Wolfgang Pitsch, Manuel Saorí and Simone Virili, this characterization only works in categories which are both (AB4) and (AB4*). \\

Let $\C$ be an abelian category and let $(\D, \E)$ be a cotorsion pair in $\C$.
We begin by proving a rather technical proposition  that we will need later to prove the completeness (in under certain hypothesis) of the induced cotorsion pair; to do so we need some preliminary algebraic results.


\begin{lemma}
    \label{lem:inv-limit-chain}
    Let us assume $\C$ to be complete and let's consider an inverse system in $\Ch(\C)$
    \[\begin{tikzcd}
        \cdots & {P_3} & {P_2} & {P_1} & {P_0.}
        \arrow[from=1-1, to=1-2]
        \arrow["{\mu_3}", from=1-2, to=1-3]
        \arrow["{\mu_2}", from=1-3, to=1-4]
        \arrow["{\mu_1}", from=1-4, to=1-5]
    \end{tikzcd}\]
    We can define a morphism $\nu \colon \prod_{i \in \N} P_i \longrightarrow \prod_{i \in \N} P_i$ induced by 
    \[\begin{tikzcd}
        {\prod_{i \in \N} P_i} & {\prod_{i \in \N} P_i} \\
        {P_{n+1}} & {P_n.}
        \arrow["\nu", from=1-1, to=1-2]
        \arrow["{\pi_{n+1}}"', from=1-1, to=2-1]
        \arrow["{\pi_n}", from=1-2, to=2-2]
        \arrow["{\mu_{n+1}}", from=2-1, to=2-2]
    \end{tikzcd}\]
    If we assume each $\mu_i$ to be epic and dimensionwise split, the following sequence is split exact in each degree.
    \[\begin{tikzcd}
    0 
    \arrow[r] 
    & \varprojlim P_n 
    \arrow[r, "\lambda"] 
    & \displaystyle\prod_{n \geq 0} P_n 
    \arrow[r, "1-\nu"] 
    & \displaystyle\prod_{n \geq 0} P_n 
    \arrow[r] 
    & 0
  \end{tikzcd}\]
\end{lemma}

\begin{proof}
    See \cite{MurfetDerived}, definition 29, lemma 67 and proposition 85.
\end{proof}

\begin{lemma}
    \label{lem:inv-limit-group}
    Let's consider an inverse system of abelian groups
    \[\begin{tikzcd}
        \cdots & {P_3} & {P_2} & {P_1} & {P_0.}
        \arrow[from=1-1, to=1-2]
        \arrow["{\mu_3}", from=1-2, to=1-3]
        \arrow["{\mu_2}", from=1-3, to=1-4]
        \arrow["{\mu_1}", from=1-4, to=1-5]
    \end{tikzcd}\]
    If each $\mu_i$ is surjective, the following sequence is exact, where $\nu$ is defined by $\pi_n \circ \nu = \mu_{n+1} \circ \pi_{n+1}$ for all $n \geq 0$ (as in the previous lemma).
    \[\begin{tikzcd}
    0 
    \arrow[r] 
    & \varprojlim P_n 
    \arrow[r, "\lambda"] 
    & \displaystyle\prod_{n \geq 0} P_n 
    \arrow[r, "1-\nu"] 
    & \displaystyle\prod_{n \geq 0} P_n 
    \arrow[r] 
    & 0
  \end{tikzcd}\]
\end{lemma}

\begin{proof}
    See \cite{Weibel_1994}, lemma 3.5.3.
\end{proof}

Before proving the following proposition, it is important to remark that results on filtered colimits/injective limits which are dual to the above lemmas also hold.

\begin{proposition}
    \label{prop:enough-objects}
    Let us assume that $\C$ is complete, cocomplete, (AB4) and (AB4*); let $X \in \Ch(\C)$.
    \begin{enumerate}
        \itemsep0em 
        \item If $(\D, \E)$ has enough $\E$-objects, then there exists an exact sequence in $\Ch(\C)$
            \[\begin{tikzcd}
                0 & E & P & X & {0,}
                \arrow[from=1-1, to=1-2]
                \arrow[from=1-2, to=1-3]
                \arrow[from=1-3, to=1-4]
                \arrow[from=1-4, to=1-5]
            \end{tikzcd}\]
            where $E \in \dg{\E}$ and $P \in \ex$.
        \item If $(\D, \E)$ has enough $\D$-objects, then there exists an exact sequence in $\Ch(\C)$
            \[\begin{tikzcd}
                0 & X & P & D & {0,}
                \arrow[from=1-1, to=1-2]
                \arrow[from=1-2, to=1-3]
                \arrow[from=1-3, to=1-4]
                \arrow[from=1-4, to=1-5]
            \end{tikzcd}\]
            where $D \in \dg{\D}$ and $P \in \ex$.
    \end{enumerate}
\end{proposition}




\begin{proof}
    We will only prove the first statement, given that the other is dual. Let us assume that $(\D, \E)$ has enough $\E$-objects and let $X \in \Ch(\C)$. We set $C_n := \coker(d_{n+1}^X)$ and let $\rho_n := X_{n} \longrightarrow C_n$ be the natural epimorphism. We will now define a new complex $Y^0$ by performing the so-called "killing boundaries construction": let us choose an integer $n_0 \in \Z$ and a monomorphism \[\iota \colon C_{n_0} \longrightarrow E_{n_0-1}\] for $E_{n_0-1} \in \E$ (there are enough $\E$-objects), so that the objects of the complex are $Y^0_{n_0-1} := X_{n-1} \oplus E_{n_0-1}$ and $Y^0_k := X_k$ ($k \neq n_0$), while the boundary  is given by 
    \[\begin{tikzcd}
        \cdots & {X_{n+1}} & {X_n} & {X_{n-1} \oplus E_{n_0-1}} & {X_{n-2}} & \cdots
        \arrow[from=1-1, to=1-2]
        \arrow["{d^X_{n+1}}", from=1-2, to=1-3]
        \arrow["{(d^X_{n}, \iota \rho_n)}", from=1-3, to=1-4]
        \arrow["{d^X_{n-1}}", from=1-4, to=1-5]
        \arrow[from=1-5, to=1-6]
    \end{tikzcd}\]
    It is clear that we get a morphism $Y^0 \longrightarrow X$ which is a degreewise split epi and whose kernel is $E^0 := S^{n_0-1}E_{n_0 -1} \in \dg{\E}$. It is easy to see that  $Y^0$ is exact at $n_0$ and that the induced morphism $B_kY^0 \longrightarrow B_k X$ is an isomorphism for $k \neq n_0$. Given a sequence $(n_0, n_1, n_2, \dots)$ such that every integer appears in it infinitely many times we can repeat the killing boundaries construction to obtain a sequence of complexes $Y^l$ and degreewise split epis $Y^{l+1} \longrightarrow Y^l$ such that $Y^l$ is exact at position $n_l$, $B_kY^{l+1} \longrightarrow B_k Y^l$ is an isomorphism for $k \neq l+1$ and \[\ker(Y^{l+1} \longrightarrow Y^l) = S^{n_l-1}E_{n_l -1} \] is in $\dg{\E}$\\
    We continue by defining, just like we did for $l=0$, the complex
    \[E^l := \ker(Y^l \longrightarrow Y^{l-1} \longrightarrow \dots \longrightarrow Y^0 \longrightarrow X)\]
    and remarking that the induced morphism $E^{l+1} \longrightarrow E^l$ must be degreewise split epis ($\ker$ is a functor, so it sends split epis in the arrow category to split epis); it is also important to remark that $E^{l+1}$ is an extension of $S^{n_l-1}E_{n_l -1}$ and $E^l$, therefore by induction it must also be in $\dg{\E}$. Furthermore for all $l\in \N$ we obtain the following degreewise split exact sequences (composition of split epis is a split epi):
    \[\begin{tikzcd}
        0 & {E^l} & {Y^l} & X & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    Glueing the sequences associated to each degree yields 
    \[\begin{tikzcd}
        0 & {\displaystyle\prod_{l= 0}^{\infty} E^l_k} & {\displaystyle\prod_{l= 0}^{\infty} Y^l_k} & {\displaystyle\prod_{l= 0}^{\infty} X_k} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    which is split exact for all $k \in \Z$, therefore
    \[\begin{tikzcd}
        0 & {\displaystyle\prod_{l= 0}^{\infty} E^l} & {\displaystyle\prod_{l= 0}^{\infty} Y^l} & {\displaystyle\prod_{l= 0}^{\infty} X} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    is exact. We continue by setting $E := \varprojlim E^l$ and $P:= \varprojlim Y^l$ (and remarking that, trivially, $X = \varprojlim X$) and applying lemma \ref{lem:inv-limit-chain} to obtain a commutative diagram with exact columns and where the last two rows are also exact;
    \[\begin{tikzcd}
        & 0 \arrow[d] & 0 \arrow[d] & 0 \arrow[d] & \\
        0 \arrow[r] 
        & E \arrow[r] \arrow[d] 
        & P \arrow[r] \arrow[d] 
        & X \arrow[r] \arrow[d] 
        & 0 \\
        0 \arrow[r] 
        & \displaystyle\prod_{l = 0} ^{\infty} E^l \arrow[r] \arrow[d, "1-\nu"] 
        & \displaystyle\prod_{l = 0} ^{\infty} Y^l \arrow[r] \arrow[d, "1-\nu"] 
        & \displaystyle\prod_{l = 0} ^{\infty} X \arrow[r] \arrow[d, "1-\nu"] 
        & 0 \\
        0 \arrow[r] 
        & \displaystyle\prod_{l = 0} ^{\infty} E^l \arrow[r] \arrow[d] 
        & \displaystyle\prod_{l = 0} ^{\infty} Y^l \arrow[r] \arrow[d] 
        & \displaystyle\prod_{l = 0} ^{\infty} X \arrow[r] \arrow[d] 
        & 0 \\
        & 0 & 0 & 0 &
    \end{tikzcd}\]
    from the nine lemma follows the exactness of the sequence
    \[\begin{tikzcd}
        0 & E & P & X & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    To prove that this sequence is the one we are looking for, it remains to prove that $E \in \dg{\E}$ and $P \in \ex$, we begin by investigating $E$. The first step is proving that $E_k \in \E$ for all $k$, but if we set $K^l := \ker(E^{l+1} \longrightarrow E^l)$, the following diagram of kernels 
    \[\begin{tikzcd}
        0 & {K^l} & {S^{n_l-1}E_{n_l-1}} & 0 & \\
        0 & {E^{l+1}} & {Y^{l+1}} & X & 0 \\
        0 & {E^l} & {Y^l} & X & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-2, to=2-2]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-2, to=3-2]
        \arrow[from=2-3, to=2-4]
        \arrow[from=2-3, to=3-3]
        \arrow[from=2-4, to=2-5]
        \arrow[from=2-4, to=3-4]
        \arrow[from=3-1, to=3-2]
        \arrow[from=3-2, to=3-3]
        \arrow[from=3-3, to=3-4]
        \arrow[from=3-4, to=3-5]
    \end{tikzcd}\]
    shows that $K^l \cong S^{n_l-1}E_{n_l-1} \in \dg{\E}$. Given that the morphism in the projective system used to compute $E$ are degreewise split epis, we can compute $E_k$ degreewise to find
    \[ E_k = \left( \varprojlim E^l \right)_k = E^0_k \oplus \prod_{l=0}^{\infty} K^l_k,\]
    thus $E_k\in \E$, given that $\E$ is part of a cotorsion pair and therefore closed under products (see lemma \ref{lem:cotorsion-prod}). We can apply the $\Hom(D, {-})$ functor (for a certain $D \in \D$) to the short exact sequence 
    \[\begin{tikzcd}
        0 & {K^l} & {E^{l+1}} & {E^l} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    to obtain 
    \[\begin{tikzcd}[sep=small]
        \cdots & {\Hom_\C(D, E^{l+1})} & {\Hom_\C(D, E^l)} & {\Ext^1_\C(D, K^l)} & {\cdots.}
	    \arrow[from=1-1, to=1-2]
	    \arrow[from=1-2, to=1-3]
	    \arrow[from=1-3, to=1-4]
	    \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    Given that $K^l \in \dg{\E}$ and $D \in \D$, we know that ${\Ext^1_\C(D, K^l)} = 0$, therefore the morphism 
    \[{\Hom_\C(D, E^{l+1})} \longrightarrow {\Hom_\C(D, E^l)}\]
    must be epic for all $l$. Applying lemma \ref{lem:inv-limit-group}, we obtain the following exact sequence:
    \[\begin{tikzcd}[sep=tiny]
        0 & {\varprojlim \Hom_\C(D,E^l)} & {\displaystyle \prod_{l=0}^{\infty}\Hom_\C(D,E^l)} & {\displaystyle \prod_{l=0}^{\infty}\Hom_\C(D,E^l)} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    hence, given that $\prod_{l=0}^{\infty}\Hom_\C(D,E^l)$ is exact ($E^l \in \dg{\E}$ and the product of exact complexes of abelian groups is exact), $\varprojlim \Hom_\C(D,E^l) \cong \Hom_\C(D,E)$ is exact ($\Hom(D, {-})$ is a continuous functor).\\
    All that remains is showing that $P \in \ex$, we will show that it is exact at each homological position $k \in \Z$. Let us choose a cofinal subsequence $(n_{m_0}, n_{m_1}, \dots)$ of $(n_0, n_1, \dots)$ such that $n_{m_t} = k$ for all $t$; such a subsequence must exist given that each integer appears infinitely many times in the original sequence, furthermore the cofinality condition implies that we can compute the limit using only complexes that are exact in position $k$:
    \[P = \varprojlim_{i \in \N} Y^i = \varprojlim_{i \in \N} Y^{m_i}.\]
    Applying lemma \ref{lem:inv-limit-chain} allows us to write the following
    \[\begin{tikzcd}
        0 & P & {\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}} & {\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow["1-\nu", from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    which is split exact in each degree and yields the following exact sequence:
    \[\begin{tikzcd}
        { 0} & {H_k(P)} & {H_k\left({\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}}\right)} & {H_k\left({\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}}\right)} & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow["{H_k(1- \nu)}", from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    If we denote as $\mu_j$ the morphism $Y^{m_j} \longrightarrow Y^{m_{j-1}}$, from the definition of $\nu$ follows immediately that \[\pi_{m_j} (1-\nu) = \pi_{m_j} - \mu_{j+1} \pi_{{m_{j+1}}}\] for all $j$; from the exactness of $Y^l$ and the fact that for abelian groups products are exact, by denoting as $\mu_{t,i}$ and $\pi_{t,i}$ the degree $i$ component of $\mu_t$ and $\pi_t$, we deduce  
    \begin{align*}
        \mu_{j+1, k} \pi_{m_{j+1}, k} \left(Z_k \prod_{i=0}^{\infty} Y^{m_i}  \right) &=  \mu_{j+1, k} \pi_{m_{j+1}, k}\left( \prod_{i=0}^{\infty} Z_k  Y^{m_i}  \right) = \\
        &= \mu_{j+1, k} \left( Z_k Y^{m_{j+1}} \right) = \\
        &= \mu_{j+1, k}  \left(B_k Y^{m_{j+1}} \right) = \\
        &= \mu_{j+1, k} d^{Y^{m_{j+1}}}_{k+1} \left(Y^{m_{j+1}}_{k+1} \right) = \\
        &=  d^{Y^{m_{j}}}_{k+1} \mu_{j+1, k+1} \left(Y^{m_{j+1}}_{k+1} \right) \subseteq B_k Y^{m_j}.
    \end{align*}
    Given that the (AB4*) hypothesis allows us to write \[\prod_{j=0}^\infty B_k Y^{m_j} =   B_k \prod_{j=0}^\infty Y^{m_j},\] we can conclude that $H_k (\mu_{j+1} \pi_{{m_{j+1}}})$ is trivial, therefore $H_k(\pi_{m_j}(1-\nu)) = H_k(\pi_j)$ for all $j$, thus $1-\nu$ acts as the identity on the $k$-th homology groups and hence $H_k(P) = 0$. Given that this holds for all $k$, $P$ must be exact.

    
\end{proof}







\begin{definition}[hereditary cotorsion pairs]
    We say that the cotorsion pair $(\D, \E)$ is \textbf{hereditary} if $\Ext^i_\C(D,E) = 0$ for all $i \geq 1$ and $D\in \D, E\in \E$.
\end{definition}

We will additionally need two technical definitions.

\begin{definition}[Covers and envelopes]
    Let $\F$ be a subclass of $\ob(\C)$, we say that a morphism $\phi \colon F \longrightarrow X$ is an $\F$\textbf{-precover} of $X$ if the sequence of abelian groups
    \[\begin{tikzcd}
        {\Hom_\C(F',F)} & {\Hom_\C(F',X)} & 0
        \arrow["{\phi_*}", from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
    \end{tikzcd}\]
    is exact for all $F' \in \F$. An $\F$-precover is said to be \textbf{special} if $\phi$ is an epimorphism and $\ker(\phi) \in \rperp{\F}$. If $\phi$ is an $\F$-precover such that any endomorphisms $f\colon F \longrightarrow F$ that satisfies $\phi f = \phi$ is necessarily an automorphism of $F$, we say that $\phi$ is an $\F$\textbf{-cover} of $X$. \\
    Dually, we say that a morphism $\psi \colon X \longrightarrow F$ is an $\F$\textbf{-preenvelope} of $X$ if the sequence of abelian groups
    \[\begin{tikzcd}
        {\Hom_\C(F,F')} & {\Hom_\C(X,F')} & 0
        \arrow["{\psi^*}", from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
    \end{tikzcd}\]
    is exact for all $F' \in \F$. An $\F$-preenvelope is said to be \textbf{special} if $\psi$ is a monomorphism and $\coker(\psi) \in \lperp{\F}$. If $\psi$ is an $\F$-preenvelope such that any endomorphism $f\colon F \longrightarrow F$ that satisfies $f\psi  = \psi$ is necessarily an automorphism of $F$, we say that $\psi$ is an $\F$\textbf{-envelope} of $X$. \\
\end{definition}

It is clear that requiring every object to have a special $\D$-precover is equivalent to asking that $(\D, \E)$ has enough projectives, while requiring every object to have a special $\E$-preenvelope is equivalent to asking that $(\D, \E)$ has enough injectives.

\begin{lemma}
    \label{lem:cover-ext}
    Let $(\D, \E)$ be a hereditary cotorsion pair in $\C$, the class of objects that admit a special $\D$-precover and the class of objects that admit a special $\E$-preenvelope are closed under extensions, i.e. for any exact sequence
    \[\begin{tikzcd}
        0 & {X_1} & X & {X_2} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    in $\C$ the following holds:
    \begin{enumerate}
        \itemsep0em 
        \item If $X_1, X_2$ have a special $\D$-precover then $X$ has a special $\D$-precover.
        \item If $X_1, X_2$ have a special $\E$-preenvelope then $X$ has a special $\E$-preenvelope.
    \end{enumerate}
\end{lemma}

\begin{proof}
    We will just prove the first statement since the second follows by duality. \\  
    Let $\phi_1\colon D_1 \longrightarrow X_1$ and $\phi_2\colon D_2 \longrightarrow X_2$ be special $\D$-precovers, we want to use these to define a special $\D$-precover of $X$. The extension $\xi$
    \[\begin{tikzcd}
        0 & {X_1} & X & {X_2} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    induces through the morphism $\phi_2$ another extension $\xi' := \phi_2^* \xi$ (see section \ref{sec:extensions}) which fits the commutative diagram with exact rows below
    \[\begin{tikzcd}
        &&& 0 & 0 & \\
        &&& {E_2} & {E_2} \\
        {\xi':} & 0 & {X_1} & Y & {D_2} & 0 \\
        {\xi:} & 0 & {X_1} & X & {X_2} & 0 \\
        &&& 0 & 0
        \arrow[from=1-4, to=2-4]
        \arrow[from=1-5, to=2-5]
        \arrow["{\id_{E_2}}", from=2-4, to=2-5]
        \arrow[from=2-4, to=3-4]
        \arrow[from=2-5, to=3-5]
        \arrow[from=3-2, to=3-3]
        \arrow[from=3-3, to=3-4]
        \arrow["{\id_{X_1}}", from=3-3, to=4-3]
        \arrow[from=3-4, to=3-5]
        \arrow[from=3-4, to=4-4]
        \arrow[from=3-5, to=3-6]
        \arrow["{\phi_2}", from=3-5, to=4-5]
        \arrow[from=4-2, to=4-3]
        \arrow[from=4-3, to=4-4]
        \arrow[from=4-4, to=4-5]
        \arrow[from=4-4, to=5-4]
        \arrow[from=4-5, to=4-6]
        \arrow[from=4-5, to=5-5]
    \end{tikzcd}\]
    where $Y := X \times_{X_2} D_2$ and $E_2 := \ker(\phi_2)$. If we now set $E_1:= \ker(\phi_1)$, we get an exact sequence 
    \[\begin{tikzcd}
        0 & {E_1} & {D_1} & {X_1} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow["{\phi_1}", from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    from which, by applying $\Hom_\C(D_2, {-})$, we get
    \[\begin{tikzcd}
        \cdots & {\Ext^1_\C(D_2,D_1)} & {\Ext^1_\C(D_2,X_1)} & {\Ext^2_\C(D_2,E_1)} & {\cdots.}
        \arrow[from=1-1, to=1-2]
        \arrow["{(\phi_1)_*}", from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    Given that $E_1 \in \E$ ($\phi_2$ is special) and $(\D, \E)$ is hereditary, ${\Ext^2_\C(D_2,E_1)} = 0$ and so ${(\phi_1)_*} \colon {\Ext^1_\C(D_2,D_1)} \longrightarrow {\Ext^1_\C(D_2,X_1)}$ is surjective, therefore there exists an extension $\xi''$
    \[\begin{tikzcd}
        {\xi'':} & 0 & {D_2} & D & {D_1} & 0
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
        \arrow[from=1-5, to=1-6]
    \end{tikzcd}\]
    such that $(\phi_1)_*(\xi'') = \xi'$; given that $\D$ is closed under extensions, $D \in \D$. By writing explicitly the action of $(\phi_1)^*$ we obtain the commutative diagram
    \[\begin{tikzcd}
        && 0 & 0 && \\
        && {E_1} & {E_1} \\
        {\xi'':} & 0 & {D_1} & D & {D_2} & 0 \\
        {\xi':} & 0 & {X_1} & Y & {D_2} & {0.}
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=2-4]
        \arrow["{\id_{E_1}}", from=2-3, to=2-4]
        \arrow[from=2-3, to=3-3]
        \arrow[from=2-4, to=3-4]
        \arrow[from=3-2, to=3-3]
        \arrow[from=3-3, to=3-4]
        \arrow["{\phi_1}", from=3-3, to=4-3]
        \arrow[from=3-4, to=3-5]
        \arrow[from=3-4, to=4-4]
        \arrow[from=3-5, to=3-6]
        \arrow["{\id_{D_2}}", from=3-5, to=4-5]
        \arrow[from=4-2, to=4-3]
        \arrow[from=4-3, to=4-4]
        \arrow[from=4-4, to=4-5]
        \arrow[from=4-5, to=4-6]
    \end{tikzcd}\]
    By joining together the two diagrams (and using the serpent's lemma with the fact that $\phi_1$ is epic) we get
    \[\begin{tikzcd}
        & 0 & {E_1} & E & {E_2} & 0 \\
        {\xi'':} & 0 & {D_1} & D & {D_2} & 0 \\
        {\xi':} & 0 & {X_1} & Y & {D_2} & 0 \\
        {\xi:} & 0 & {X_1} & X & {X_2} & {0,}
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=1-5]
        \arrow[from=1-4, to=2-4]
        \arrow[from=1-5, to=1-6]
        \arrow[from=1-5, to=2-5]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-3, to=2-4]
        \arrow["{\phi_1}", from=2-3, to=3-3]
        \arrow["{\phi_1}"'{pos=0.2}, curve={height=12pt}, from=2-3, to=4-3]
        \arrow[from=2-4, to=2-5]
        \arrow[from=2-4, to=3-4]
        \arrow[from=2-5, to=2-6]
        \arrow["{\id_{D_2}}", from=2-5, to=3-5]
        \arrow["{\phi_2}"'{pos=0.2}, curve={height=12pt}, from=2-5, to=4-5]
        \arrow[from=3-2, to=3-3]
        \arrow[from=3-3, to=3-4]
        \arrow["{\id_{X_1}}", from=3-3, to=4-3]
        \arrow[from=3-4, to=3-5]
        \arrow[from=3-4, to=4-4]
        \arrow[from=3-5, to=3-6]
        \arrow["{\phi_2}", from=3-5, to=4-5]
        \arrow[from=4-2, to=4-3]
        \arrow[from=4-3, to=4-4]
        \arrow[from=4-4, to=4-5]
        \arrow[from=4-5, to=4-6]
    \end{tikzcd}\]
    where $E$ is the kernel of the composition $\phi \colon D \longrightarrow Y \longrightarrow X$; given that $\E$ is closed under extensions, $E \in \E$ and therefore $\phi_*$ is surjective (the obstructing $\Ext$ vanishes because $(\D, \E)$ is a cotorsion pair). We conclude by remarking that both $D \longrightarrow Y$ and $Y \longrightarrow X$ are epic (they are pullbacks/pushouts of epic morphisms), therefore $\phi\colon D \longrightarrow X$ is a special $\D$-precover.
\end{proof}

\begin{remark}
    It is important to remark that in the proof above we have actually shown something stronger that the thesis: given special $\D$-precovers $\phi_1 \colon D_1 \longrightarrow X_1$ and $\phi_2\colon D_2 \longrightarrow X_2$ we can actually choose a special $\D$-precover $\phi\colon D \longrightarrow X$ such that the following diagram with exact rows and columns commutes
    \[\begin{tikzcd}

        & 0 & 0 & {0,}
        \arrow[from=1-2, to=2-2]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-2, to=3-2]
        \arrow[from=2-3, to=2-4]
        \arrow[from=2-3, to=3-3]
        \arrow[from=2-4, to=2-5]
        \arrow[from=2-4, to=3-4]
        \arrow[from=3-1, to=3-2]
        \arrow[from=3-2, to=3-3]
        \arrow["{\phi_1}", from=3-2, to=4-2]
        \arrow[from=3-3, to=3-4]
        \arrow["\phi", from=3-3, to=4-3]
        \arrow[from=3-4, to=3-5]

\begin{lemma}
    \label{lem:find-precover}
    If $(\D, \E)$ is complete and hereditary, then every exact complex admits a special $\DD$-precover and a special $\EE$-preenvelope.
\end{lemma}

\begin{proof}
    Let $X\in \ex$ be an exact complex, therefore for any $k \in \Z$ there exist an exact sequence 
    \[\begin{tikzcd}
        0 & {Z_{k}X} & {X_k} & {Z_{k-1}X} & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    Given that $(\D, \E)$ is complete we can find special $\D$-precovers $\phi_i' \colon D_i'  \longrightarrow Z_i X$ for all $i\in \Z$, so we can apply lemma \ref{lem:cover-ext} and the  observation following it to find a commutative diagram 
    \[\begin{tikzcd}
        & 0 & 0 & 0 & \\
        0 & {\ker(\phi_k')} & {\ker(\phi_k)} & {\ker(\phi_{k-1}')} & 0 \\
        0 & {D_k'} & {D_k} & {D_{k-1}'} & 0 \\
        0 & {Z_{k}X} & {X_k} & {Z_{k-1}X} & 0 \\
        & 0 & 0 & 0
        \arrow[from=1-2, to=2-2]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-2, to=3-2]
        \arrow[from=2-3, to=2-4]
        \arrow[from=2-3, to=3-3]
        \arrow[from=2-4, to=2-5]
        \arrow[from=2-4, to=3-4]
        \arrow[from=3-1, to=3-2]
        \arrow["{\delta_k}", from=3-2, to=3-3]
        \arrow["{\phi_k'}", from=3-2, to=4-2]
        \arrow["{\varepsilon_k}", from=3-3, to=3-4]
        \arrow["{\phi_k}", from=3-3, to=4-3]
        \arrow[from=3-4, to=3-5]
        \arrow["{\phi_{k-1}'}", from=3-4, to=4-4]
        \arrow[from=4-1, to=4-2]
        \arrow[from=4-2, to=4-3]
        \arrow[from=4-2, to=5-2]
        \arrow[from=4-3, to=4-4]
        \arrow[from=4-3, to=5-3]
        \arrow[from=4-4, to=4-5]
        \arrow[from=4-4, to=5-4]
    \end{tikzcd}\]
    such that $\phi_k$ is a special $\D$-precover. Setting $d^D_k= \delta_{k-1}\varepsilon_k \colon D_k \longrightarrow D_{k-1}$ allows us to define the complex $D= \{ (D_k, d^D_k)\}_k$, along with the morphism of complexes $\phi=\{\phi_k\}_k \colon D \longrightarrow X$. We can see for all $k$ that $D_k \in \D$, that \[\ker(d^D_k) = \ker(\delta_{k-1}\varepsilon_k) = \ker(\varepsilon_k) = D'_k\] and that \[\im(d^D_{k+1}) = \im(\delta_{k}\varepsilon_{k+1}) = \im(\delta_{k}) = D'_k,\] therefore $D \in \DD$; athe same exact argument shows that the complex $\ker(\phi)$ is in $\EE$.
    The construction above makes also clear that $\phi_k$ must be epic, therefore $\phi$ is the special $\DD$-precover we are looking for. The case of special $\EE$-preenvelopes is dual.

\end{proof}


We are finally ready to prove some of the main result of this thesis: a characterization of completeness and compatibility of the induced cotorsion pair.



\begin{theorem}
    Let $(\D, \E)$ be complete and hereditary and let $\C$ be complete, cocomplete, (AB4) and (AB4*); then the induced cotorsion pairs $(\DD, \dg{\E})$ and $(\dg{\D}, \EE)$ are complete. 
\end{theorem}

\begin{proof}
    Let $X \in \Ch(\C)$, by applying proposition \ref{prop:enough-objects} we can find an exact sequence
    \[\begin{tikzcd}
        0 & E & P & X & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    with $E \in \dg{\E}$ and $P \in \ex$. Applying lemma \ref{lem:find-precover} to the exact complex $P$, we find a special $\DD$ precover $\phi$ of $P$ as in the sequence below:
    \[\begin{tikzcd}
        0 & {E'} & D & P & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow["\phi", from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    By defining $E'':= E \times_P D$ we obtain a pullback commutative diagram as follows:
    \[\begin{tikzcd}
        & 0 & 0 && \\
        & {E'} & {E'} \\
        0 & {E''} & D & X & 0 \\
        0 & E & P & X & 0 \\
        & 0 & {0.}
        \arrow[from=1-2, to=2-2]
        \arrow[from=1-3, to=2-3]
        \arrow["{\id_{E'}}", from=2-2, to=2-3]
        \arrow[from=2-2, to=3-2]
        \arrow[from=2-3, to=3-3]
        \arrow[from=3-1, to=3-2]
        \arrow[from=3-2, to=3-3]
        \arrow[from=3-2, to=4-2]
        \arrow[from=3-3, to=3-4]
        \arrow["\phi"', from=3-3, to=4-3]
        \arrow[from=3-4, to=3-5]
        \arrow["{\id_X}"', from=3-4, to=4-4]
        \arrow[from=4-1, to=4-2]
        \arrow[from=4-2, to=4-3]
        \arrow[from=4-2, to=5-2]
        \arrow[from=4-3, to=4-4]
        \arrow[from=4-3, to=5-3]
        \arrow[from=4-4, to=4-5]
    \end{tikzcd}\]
    Given that $E''$ is an extension of $E$ and $E'$, $E'' \in \dg{\E}$ and that $D \in \DD$, the second row of this diagram shows that $(\DD, \dg{\E})$ has enough projectives. \\
    We now continue by using the completeness of $(\D, \E)$ to find a monomorphism $\iota_k\colon X_k \longrightarrow E_k'$ for all $k \in \Z$ with $E_k' \in \E$. By lemma \ref{lem:dg-sphere} the complexes $D^{k+1}E_k'$ are in $\dg{\E}$, therefore, by lemma \ref{lem:cotorsion-prod}, $\prod_{k \in \Z} D^{k+1}E_k'$ also is. Thus, by defining the monomorphism $\psi \colon X \longrightarrow \prod_{k \in \Z} D^{k+1}E_k'$ through
    \[\begin{tikzcd}
        {} & {\psi_k\colon X_k} && {\left(\displaystyle\prod_{k \in \Z} D^{k+1}E_k' \right)_k = E_k'\times E_{k-1}'}
        \arrow["{(\iota_k, \iota_{k-1}d^X_k)}", from=1-2, to=1-4]
    \end{tikzcd}\]
    and by setting $Y := \coker(\psi)$, we can  find a short exact sequence 
    \[\begin{tikzcd}
        0 & X & {\prod_{k \in \Z} D^{k+1}E_k'} & Y & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow["\psi", from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    If we now apply to $Y$ the fact that $(\DD, \dg{\E})$ has enough projectives (which we have already proven), we can find another exact sequence
    \[\begin{tikzcd}
        0 & {E''} & D & Y & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    hence by defining $E := {\prod_{k \in \Z} D^{k+1}E_k'} \times_Y D$ we get a pullback diagram
    \[\begin{tikzcd}
        && 0 & 0 & \\
        && {E''} & {E''} \\
        0 & X & E & D & 0 \\
        0 & X & {\prod_{k \in \Z} D^{k+1}E_k'} & Y & 0 \\
        & {} & 0 & 0
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=2-4]
        \arrow["{\id_{E''}}", from=2-3, to=2-4]
        \arrow[from=2-3, to=3-3]
        \arrow[from=2-4, to=3-4]
        \arrow[from=3-1, to=3-2]
        \arrow[from=3-2, to=3-3]
        \arrow["{\id_X}"', from=3-2, to=4-2]
        \arrow[from=3-3, to=3-4]
        \arrow[from=3-3, to=4-3]
        \arrow[from=3-4, to=3-5]
        \arrow[from=3-4, to=4-4]
        \arrow[from=4-1, to=4-2]
        \arrow["\psi", from=4-2, to=4-3]
        \arrow[from=4-3, to=4-4]
        \arrow[from=4-3, to=5-3]
        \arrow[from=4-4, to=4-5]
        \arrow[from=4-4, to=5-4]
    \end{tikzcd}\]
    whose first exact row is the proof that $(\DD, \dg{\E})$ has enough injectives.
    The other statements follow by duality.

\end{proof}

\begin{theorem}
    \label{thm:induced-compatible}
    Let $(\D, \E)$ be hereditary. If $(\D, \E)$ has enough $\D$-objects, then $\dg{\E} \cap \ex = \EE$ and if $(\D, \E)$ has enough $\E$-objects, then $\dg{\D} \cap \ex = \DD$. In particular, if $(\D, \E)$ is complete then the induced cotorsion pairs are compatible.
\end{theorem}

\begin{proof}
    We will only prove $\dg{\E} \cap \ex = \EE$, the other case is dual. \\
    We begin by remarking that we have already proven that $\EE \subseteq \dg{\E} \cap \ex$, so we only need to prove the opposite containment. Let $X$ be an exact complex in $\dg{\E}$, we want to prove that $Z_k X \in \E$ for all $k \in \Z$, i.e. that $\Ext^1_\C(D_0, Z_k X)=0$ for all $D_0 \in \D$. Given that the exactness of $X$ allows us to write the short exact sequence 
    \[\begin{tikzcd}
        0 & {Z_{k}X} & {X_k} & {Z_{k-1}X} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    the vanishing of this $\Ext$ functor is equivalent to asking that the morphism
    \[\Hom_\C(D_0, X_k)\longrightarrow \Hom_\C(D_0, Z_{k-1} X)\]
    is surjective, i.e. that all morphisms $f\colon D_0 \longrightarrow Z_{k-1} X$ lift to $X_k$. 
    By setting $Y_1 := X_k \times_{Z_{k-1}X} D_0$ we get a pullback diagram as follows.
    \[\begin{tikzcd}
        0 & {Z_kX} & {Y_1} & {D_0} & 0 \\
        0 & {Z_{k}X} & {X_k} & {Z_{k-1}X} & {0}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow["{\id_{Z_kX}}", from=1-2, to=2-2]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=1-5]
        \arrow["f", from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-3, to=2-4]
        \arrow[from=2-4, to=2-5]
    \end{tikzcd}\]
    If we now use explicitly the fact that $(\D, \E)$ has enough $\D$-objects, there must be an epimorphism $D_1 \longrightarrow Y_1$, so by composing the necessary morphisms with the diagram above we can find the following commutative diagram:
    \[\begin{tikzcd}
        0 & {\ker\left(d^D_{1}\right)} & {D_1} & {D_0} & 0 \\
        0 & {Z_{k}X} & {X_k} & {X_{k-1}.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-2, to=2-2]
        \arrow["{d^D_{1}}", from=1-3, to=1-4]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=1-5]
        \arrow[from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow["{d^X_k}", from=2-3, to=2-4]
    \end{tikzcd}\]
    This process can be repeated inductively: from a diagram
    \[\begin{tikzcd}
        0 & {Z_{n+1}X} & {X_{n+1}} & {Z_{n}X} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    and a morphism $\ker\left(d^D_{n-k+1}\right) \longrightarrow Z_{n}X$ we can get a pullback diagram
    \[\begin{tikzcd}
        0 & {Z_{n+1}X} & {Y_{n-k+2}} & {\ker\left(d^D_{n-k+1}\right)} & 0 \\
        0 & {Z_{n+1}X} & {X_{n+1}} & {Z_{n}X} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-2, to=2-2]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=1-5]
        \arrow[from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-3, to=2-4]
        \arrow[from=2-4, to=2-5]
    \end{tikzcd}\]
    which induces, by choosing an epimorphism $D_{n-k+1} \longrightarrow Y_{n-k+1}$ with $D_{n-k+1} \in \D$ and by taking the appropriate compositions, the diagram
    \[\begin{tikzcd}
        0 & {\ker\left(d^D_{n-k+2}\right)} & {D_{n-k+2}} & {D_{n-k+1}} \\
        0 & {Z_{n+1}X} & {X_{n+1}} & {X_n.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-2, to=2-2]
        \arrow["{d^D_{n-k+2}}", from=1-3, to=1-4]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-3, to=2-4]
    \end{tikzcd}\]
    Given that by construction \[\ker(d_n^D) =\im(Y_{n+1} \longrightarrow \ker(d_n^D))= \im(d^D_{n+1}),\]  we have thus found an exact complex $D$
    \[\begin{tikzcd}
        \cdots & {D_2} & {D_1} & {D_0} & 0 & 0 & {\cdots,}
        \arrow[from=1-1, to=1-2]
        \arrow["{d^D_2}", from=1-2, to=1-3]
        \arrow["{d^D_1}", from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
        \arrow[from=1-5, to=1-6]
        \arrow[from=1-6, to=1-7]
    \end{tikzcd}\]
    and a morphism $D \longrightarrow X$ such that its component in degree $0$ is exactly $f$.
    We will now prove by induction that $Z_nD \in \D$ for all $n$ (and thus $D \in \DD$): the base step is trivial, $Z_0 D =D_0 \in \D$; by applying $\Hom({-}, E)$, for some $E\in \E$, to the short exact sequence
    \[\begin{tikzcd}
        0 & {Z_nD} & {D_n} & {Z_{n-1}D} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    we obtain the exact sequence
    \[\begin{tikzcd}[sep=tiny]
        \cdots & {\Ext^1_\C(D_n, E)} & {\Ext^1_\C(Z_nD, E)} & {\Ext^2_\C(Z_{n-1}D, E)} & {\cdots,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    but $D_n \in D$ implies that ${\Ext^1_\C(D_n, E)}=0$ and the inductive hypothesis plus the fact that the cotorsion pair is hereditary implies that ${\Ext^2_\C(Z_{n-1}D, E)}= 0$, therefore ${\Ext^1_\C(Z_nD, E)} = 0$ and $Z_n D \in \D$.
    What we have just shown (and the fact that shifting degree preserves membership in $\DD$) implies that $\Hom(\Sigma ^{k-1}D,X)$ must be exact and hence morphism we defined (which is a degree $k-1$ chain map) is nullhomotopic: this, in degree zero, implies exactly that the lift we are looking for exists (if $h$ is the homotopy then $f=d^X_k h$).
\end{proof}




Given that the need to check the vanishing of all $\Ext^k$ makes definition of hereditary cotorsion pair unwieldy, we need to find a simpler equivalent condition to more easily check the hypothesis of the theorems above; luckily the following proposition holds.

\begin{proposition}
    \label{prop:hereditary-equiv}
    Let $(\D, \E)$ be a complete cotorsion pair in $\C$, the following are equivalent:
    \begin{enumerate}
        \itemsep0em 
        \item $(\D, \E)$ is hereditary, i.e. $\Ext^i_\C(D,E) = 0$ for all $i\geq 1$, $D\in\D$ and $E\in \E$;
        \item $\Ext^2_\C(D,E) = 0$ for all $D\in\D$ and $E\in \E$;
        \item $\D$ is closed under taking kernels of epimorphisms;
        \item $\E$ is closed under taking cokernels of monomorphisms.
    \end{enumerate}
\end{proposition}

\begin{proof}
        It is clear that $(1) \Rightarrow (2)$.\\
         To prove $(2) \Rightarrow (3)$, we begin by letting $D_1, D_2 \in \D$ and letting $\phi\colon D_1 \longrightarrow D_2 $ be an epimorphism, then we set $X:=\ker(\phi)$; the exact sequence 
        \[\begin{tikzcd}
            0 & X & {D_1} & {D_2} & 0
            \arrow[from=1-1, to=1-2]
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-3, to=1-4]
            \arrow[from=1-4, to=1-5]
        \end{tikzcd}\]
        induces for all $E \in \E$
        \[\begin{tikzcd}
            \cdots & {\Ext^1_\C(D_1, E)} & {\Ext^1_\C(X, E)} & {\Ext^2_\C(D_2,E)} & {\cdots,}
            \arrow[from=1-1, to=1-2]
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-3, to=1-4]
            \arrow[from=1-4, to=1-5]
        \end{tikzcd}\]
        given that ${\Ext^1_\C(D_1, E)} = 0 = {\Ext^2_\C(D_2,E)}$, it is clear that ${\Ext^1_\C(X, E)} = 0$ and $X \in \D$.\\
        Now, to prove $(3) \Rightarrow (1)$, we consider a representative $\xi$ of an element of $\Ext^n_\C(D,E)$ for $n \geq 1, D \in \D$ and $E \in E$ 
        \[\begin{tikzcd}
            0 & E & {X_{n-1}} & {X_{n-2}} & \cdots & {X_0} & D & 0
            \arrow[from=1-1, to=1-2]
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-3, to=1-4]
            \arrow[from=1-4, to=1-5]
            \arrow[from=1-5, to=1-6]
            \arrow[from=1-6, to=1-7]
            \arrow[from=1-7, to=1-8]
        \end{tikzcd}\]
        We now define another exact complex by repeating the same process of successives pullbacks we used in the proof of theorem \ref{thm:induced-compatible} to the complex $X$ given by
        \[\begin{tikzcd}
            \cdots & 0 & {X_{n-1}} & {X_{n-2}} & \cdots & {X_0} & 0 & {\cdots,}
            \arrow[from=1-1, to=1-2]
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-3, to=1-4]
            \arrow[from=1-4, to=1-5]
            \arrow[from=1-5, to=1-6]
            \arrow[from=1-6, to=1-7]
            \arrow[from=1-7, to=1-8]
        \end{tikzcd}\]
        starting from an epimorphism $P_0 \longrightarrow X_0$ with $P_0 \in \D$ (it exists thanks to the completeness of the cotorsion pair), to obtain
        \[\begin{tikzcd}
            0 & {\ker(d^P_1)} & {P_1} & {P_0} \\
            0 & {Z_1X} & {X_1} & {X_0}
            \arrow[from=1-1, to=1-2]
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-2, to=2-2]
            \arrow["{{d_1^P}}", from=1-3, to=1-4]
            \arrow[from=1-3, to=2-3]
            \arrow[from=1-4, to=2-4]
            \arrow[from=2-1, to=2-2]
            \arrow[from=2-2, to=2-3]
            \arrow[from=2-3, to=2-4]
        \end{tikzcd}\]
        where the morphisms from $P_1 \in \D$ are defined from an epimorphsim $P_1 \longrightarrow X_1 \times_{X_0} P_0$. By adding preceding zeroes to $\xi$ to form a complex $X$, we can continue the process to get a semi-infinite complex $P$ and a morphism $P \longrightarrow X$,
        \[\begin{tikzcd}
            \cdots & {P_n} & {P_{n-1}} & {P_{n-2}} & \cdots & {P_0} & 0 \\
            \cdots & 0 & {X_{n-1}} & {X_{n-2}} & \cdots & {X_0} & 0
            \arrow[from=1-1, to=1-2]
            \arrow["{{d_n^P}}", from=1-2, to=1-3]
            \arrow[from=1-2, to=2-2]
            \arrow["{{d_{n-1}^P}}", from=1-3, to=1-4]
            \arrow[from=1-3, to=2-3]
            \arrow["{{d_{n-2}^P}}", from=1-4, to=1-5]
            \arrow[from=1-4, to=2-4]
            \arrow[from=1-5, to=1-6]
            \arrow[from=1-6, to=1-7]
            \arrow[from=1-6, to=2-6]
            \arrow[from=2-1, to=2-2]
            \arrow[from=2-2, to=2-3]
            \arrow[from=2-3, to=2-4]
            \arrow[from=2-4, to=2-5]
            \arrow[from=2-5, to=2-6]
            \arrow[from=2-6, to=2-7]
        \end{tikzcd}\]
        By construction it is clear that $H_0(P) = \coker(X_1 \longrightarrow X_0) = D = H_0(X)$ and that $H_{n-1}(P) = \ker(X_{n-1} \longrightarrow X_{n-2}) = E = H_{n-1}(X)$, hence the morphism is a quasi-isomorphism and induces the following commutative diagram 
        \[\begin{tikzcd}[sep=small]
            {\xi'\colon} & 0 & E & {\coker(d^P_n)} & {P_{n-2}} & \cdots & {P_0} & D & 0 \\
            {\xi\colon} & 0 & E & {X_{n-1}} & {X_{n-2}} & \cdots & {X_0} & D & {0,}
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-3, to=1-4]
            \arrow["{{\id_E}}", from=1-3, to=2-3]
            \arrow["{{{d_{n-1}^P}}}", from=1-4, to=1-5]
            \arrow[from=1-4, to=2-4]
            \arrow["{{{d_{n-2}^P}}}", from=1-5, to=1-6]
            \arrow[from=1-5, to=2-5]
            \arrow[from=1-6, to=1-7]
            \arrow[from=1-7, to=1-8]
            \arrow[from=1-7, to=2-7]
            \arrow[from=1-8, to=1-9]
            \arrow["{{\id_D}}", from=1-8, to=2-8]
            \arrow[from=2-2, to=2-3]
            \arrow[from=2-3, to=2-4]
            \arrow[from=2-4, to=2-5]
            \arrow[from=2-5, to=2-6]
            \arrow[from=2-6, to=2-7]
            \arrow[from=2-7, to=2-8]
            \arrow[from=2-8, to=2-9]
        \end{tikzcd}\]
        which is an equivalence between extensions $\xi$ and $\xi'$.
        The same inductive argument we used in theorem \ref{thm:induced-compatible} shows that $\im(d^P_i) \in \D$ for all $i$, thus 
        \[\begin{tikzcd}
            0 & E & {\coker(d^P_n)} & {\im(d^P_n)} & 0 
            \arrow[from=1-1, to=1-2]
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-3, to=1-4]
            \arrow[from=1-4, to=1-5]
        \end{tikzcd}\]
        splits ($\Ext^1_\C(\im(d^P_n), E) = 0$); it allows us to find a retraction $\rho \colon {\coker(d^P_n)} \longrightarrow E$. We can hence write the diagram
        \[\begin{tikzcd}[sep=small]
            {\xi'\colon} & 0 & E & {\coker(d^P_n)} & {P_{n-2}} & \cdots & {P_2} & {P_1} & D & 0 \\
            {\xi''\colon} & 0 & E & E & 0 & \cdots & 0 & D & D & {0}
            \arrow[from=1-2, to=1-3]
            \arrow[from=1-3, to=1-4]
            \arrow["{\id_E}", from=1-3, to=2-3]
            \arrow["{{{d_{n-1}^P}}}", from=1-4, to=1-5]
            \arrow["\rho", from=1-4, to=2-4]
            \arrow[from=1-5, to=1-6]
            \arrow[from=1-5, to=2-5]
            \arrow[from=1-6, to=1-7]
            \arrow[from=1-7, to=1-8]
            \arrow[from=1-7, to=2-7]
            \arrow[from=1-8, to=1-9]
            \arrow[from=1-8, to=2-8]
            \arrow[from=1-9, to=1-10]
            \arrow["{\id_D}", from=1-9, to=2-9]
            \arrow[from=2-2, to=2-3]
            \arrow["{\id_E}", from=2-3, to=2-4]
            \arrow[from=2-4, to=2-5]
            \arrow[from=2-5, to=2-6]
            \arrow[from=2-6, to=2-7]
            \arrow[from=2-7, to=2-8]
            \arrow["{\id_D}"', from=2-8, to=2-9]
            \arrow[from=2-9, to=2-10]
        \end{tikzcd}\]
        that shows that $\xi'$ is equivalent to $\xi''$, but it is clear that $\xi''$ represents $0$ in $\Ext^n_\C(D,E)$, thus $\Ext^n_\C(D,E)=0$. \\
        The remaining implications $(2) \Rightarrow (4)$ and $(4) \Rightarrow (1)$ follow from duality.

\end{proof}

From this proposition follow some of the core results.

\begin{corollary}
    Let $(\D, \E)$ be complete, the following holds:
    \begin{enumerate}
        \itemsep0em 
        \item If $(\D, \E)$ is hereditary, then the induced cotorsion pairs $(\DD, \dg{\E})$ and $(\dg{\D}, \EE)$ in $\Ch(\C)$ are also hereditary.
        \item If either of the induced cotorsion pairs $(\DD, \dg{\E})$ and $(\dg{\D}, \EE)$ is hereditary, then $(\D, \E)$ is hereditary.
        \item If $\C$ is complete and cocomplete, the following statement are equivalent:
        \begin{enumerate}
            \itemsep0em
            \item $(\D, \E)$ is hereditary.
            \item $(\DD, \dg{\E})$ and $(\dg{\D}, \EE)$ are hereditary.
            \item $(\DD, \dg{\E})$ and $(\dg{\D}, \EE)$ are compatible.
        \end{enumerate}
    \end{enumerate}
\end{corollary}

\begin{proof}
    ($1$) Let's begin by proving that $(\dg{\D}, \EE)$ is hereditary by showing that $\dg{\D}$ is closed under taking kernels of epimorphism: we consider an exact sequence 
    \[\begin{tikzcd}
        0 & {D} & {D'} & {D''} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    where $D', D'' \in \dg{\D}$; given that $(\D, \E)$ is hereditary and $D_k$ is the kernel of the morphism between $D'_k\in \D $ and $D''_k \in \D$, it is clear that $D_k \in \D$ for all $k$. If we choose a complex $E \in \EE$, we get an exact sequence
    \[\begin{tikzcd}
        0 & {\Hom_\C(D_{k+n},E_n)} & {\Hom_\C(D'_{k+n},E_n)} & {\Hom_\C(D''_{k+n},E_n)} \\
        & {\Ext^1_\C(D_{k+n},E_n)} & {\Ext^1_\C(D'_{k+n},E_n)} & {\cdots,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-3, to=2-4]
    \end{tikzcd}\]
    given that for all $k,n \in \Z$ we know that $\Ext^1_\C(D_{k+n},E_n)= 0$ and in the category of abelian groups products are exact, we get an exact sequence 
    \[\begin{tikzcd}[sep=tiny]
        0 & {\displaystyle \prod _{k \in \Z}\Hom_\C(D_{k+n},E_n)} & {\displaystyle \prod _{k \in \Z}\Hom_\C(D'_{k+n},E_n)} & {\displaystyle \prod _{k \in \Z}\ \Hom_\C(D''_{k+n},E_n)} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    thus the following sequence is exact.
    \[\begin{tikzcd}[sep=small]
        0 & {\Hom(D,E)} & {\Hom(D',E)} & {\Hom(D'',E)} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    From the associated long exact sequence is clear that $\Hom(D,E)$ is exact, therefore $D \in \dg{\D}$. The other case is dual. \\ 
    (2) Let us assume that $(\DD, \dg{\E})$ is hereditary; we consider an exact sequence in the base category $\C$ as follows
    \[\begin{tikzcd}
        0 & X & {X'} & {X''} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    with $X',X'' \in \D$, we need to show that $X \in \D$. It is clear that the complexes $D^0 X'$ and $D^0 X''$ are in $\DD$, therefore using the following exact sequence
    \[\begin{tikzcd}
        0 & {D^0X} & {D^0X'} & {D^0X''} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    and the hereditariness of $(\DD, \dg{\E})$ (in the form of proposition \ref{prop:hereditary-equiv}), we know that $D^0X \in \DD$; this implies $X \in \D$. \\
    (3) The fact that condition (a) implies condition (c) is exactly theorem \ref{thm:induced-compatible}, while the fact that condition (a) is equivalent to condition (b) is given by points (1) and (2) of this corollary; thus we only need to show that condition (c) implies condition (a). We will show that $\DD = \dg{\D} \cap \ex$ implies that $(\D, \E)$ is hereditary. Let us consider an exact sequence
    \[\begin{tikzcd}
        0 & D & {D'} & {D''} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    with $D',D'' \in \D$. Given that $(\D, \E)$ is complete (in particular it has enough projectives) we can find an exact complex
    \[\begin{tikzcd}
        {X\colon} & \cdots & {X_1} & {X_0} & {D'} & {D''} & 0 & \cdots
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
        \arrow[from=1-5, to=1-6]
        \arrow[from=1-6, to=1-7]
        \arrow[from=1-7, to=1-8]
    \end{tikzcd}\]
    with $X_k \in \D$ for all $k$. By applying lemma \ref{lem:dg-sphere}, we find that $X \in \dg{\D}$. By construction $X \in \ex$, so the compatibility condition implies $X \in \DD$; it follows that $Z_0X =\ker(D' \longrightarrow D'') = D \in \D$.
\end{proof}

We can finally put together all that we have proven thus far.

\begin{corollary}
    If $\C$ is complete, cocomplete, (AB4) and (AB4*) and if $(\D, \E)$ is complete and hereditary then there exists a model structure on $\Ch(\C)$ where:
    \begin{itemize}
        \itemsep0em 
        \item $\ex$ is the class of trivial objects.
        \item $\dg{\D}$ is the class of cofibrant objects.
        \item $\DD$ is the class of acyclic cofibrant objects.
        \item $\dg{\E}$ is the class of fibrant objects.
        \item $\EE$ is the class of acyclic fibrant objects.
    \end{itemize}
    Conversely, if there exists an abelian model structure on $\Ch(C)$ with trivial objects $\ex$, cofibrant objects $\dg{\D}$ and fibrant objects $\dg{\E}$, then $(\D, \E)$ must be complete and hereditary.
\end{corollary}


\section{Some examples: derived categories}

We will proceed to produce some standard examples.





\begin{lemma}
    If ($\dg{\D}$, $\ex$, $\dg{\E}$) is an Hovey triple, then a morphism is a weak equivalence in the induced model structure if and only if it is a quasi-isomorphism.
\end{lemma}

\begin{proof}
    Let $f\colon X \longrightarrow Y$ be a weak equivalence in the induced model structure, so by definition it can be written as $qi$, where $q$ is an acyclic fibration (i.e. an epi with fibrant kernel) and $i$ is an acyclic cofibration (i.e. a mono with cofibrant cokernel).
\end{proof}


\begin{definition}[Derived category]
    Let $\C$ be an abelian category and let $H$ be the class of quasi-isomorphisms of $\Ch(\C)$; the derived category of $\C$ is the localization $\Ch(\C)[H^{-1}]$.
\end{definition}
