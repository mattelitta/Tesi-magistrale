\chapter{Cotorsion pairs and chain complexes}

In the previous chapter we have seen the various interaction between cotorsion pairs and model structures on Grothendieck categories, we will now see how we can use this knowledge to study chain complexes in these categories. The theory has been deeveloped by James Gillespie in \cite{Gillespie2004Flat} and \cite{Gillespie2006Quillen} and by Xiaoyan Yang and Nanqing Ding in \cite{DingYang2014}.


\section{Initial remarks}

We begin by setting a standard notation for this chapter.

\begin{notation}
    Let $\C$ be an abelian category, let $A \in \ob(\C)$ and let $n \in \Z$. We denote as $S^n A$ the object of $\Ch(\C)$ given by the complex whose only nonzero element is $A$ in position $n$; we also denote as $D^n A$ the object of $\Ch(\C)$ given by the complex whose only nonzero elements are $A$ in position $n-1, n$  and whose only nonzero morphsim is $d_{n-1} = \id_A$.\\ 
    If $X = (X_n, d_n)$ is a chain complex in $\Ch(\C)$, we denote as $Z_nX$ the object in $\C$ given by $\ker d_n$, while we denote as $B_nX$ the object in $\C$ given by $\im d_{n+1}$; we also denote $Z_nX/B_nX$, the $n$-th homology of the complex, as $H_n X$.
    Finally, letting $Y$ be another complex, we denote as $\Hom(X, Y)$ the complex 
    \[\begin{tikzcd}
        \cdots & {\prod_{k \in \Z}\Hom_\C(X_k, Y_{k+n})} & {\prod_{k \in \Z}\Hom_\C(X_k, Y_{k+n-1})} & {\cdots,}
        \arrow[from=1-1, to=1-2]
        \arrow["{\delta_n}", from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
    \end{tikzcd}\]
    where, given $f \in {\prod_{k \in \Z}\Hom_\C(X_k, Y_{k+n})}$, we set $(\delta_n f)_k =  d^Y_{n+k} f_k - (-1)^n f_{k-1}d^X_k$.  \\
    Lastly, we define the complex $\Sigma^n X$ as the complex whose objects are $(\Sigma^nX)_k = X_{n-k}$ and whose differential is $d^{\Sigma^n X}_k = (-1)^n d^X_{n-k}$.
\end{notation}



We continue by proving some standard results about categories of chain complexes.


\begin{definition}[Degreewise Ext]
    Let $\C$ be an abelian category, let $X,Y$ be objects in $\Ch(\C)$, we define the \textbf{degreewise Ext group} $\Ext^1_{dw}(X, Y)$ as the subgroup of $\Ext^1_{\Ch(\C)}(X, Y)$ given by all extensions which are split degreewise. 
\end{definition}

\begin{lemma}
    \label{lem:deg-ext}
    Let $\C$ be an abelian category, let $X,Y$ be objects in $\Ch(\C)$. We have that
    \[\Ext^1_{dw}(X, \Sigma^{-n-1} Y) \cong H_n\Hom(X,Y) \cong  \Hom_{\Ch(\C)}(X, \Sigma^{-n} Y) / \sim,\]
    where $\sim$ denotes chain homotopy.
\end{lemma}

\begin{proof}
    Let $f=(f_k)_k$ be an element of $\Hom(X,Y)_n$. We see that $f \in \ker \delta_n$ if and only if $d^Y_{n+k} f_k = (-1)^n f_{k-1}d^X_k $, but $(-1)^n d^Y_{n+k}$ is exactly $d^{\Sigma^{-n} Y}_k$, therefore the condition is equivalent to checking if $f$ is a morphism between $X$ and $\Sigma^{-n} Y$. Similarly, the elements of $\im \delta_{n+1}$ are of the form 
    \[d^Y_{k+n+1}f_k - (-1)^{n+1}f_{k-1}d^X_k = (-1)^n d^{\Sigma^{-n} Y}_{k+1}f_k + (-1)^n f_{k-1}d^X_k,\]
    thus they are exactly null-homotopic maps from $X$ to $\Sigma^{-n}Y$; therefore $H_n\Hom(X,Y) \cong  \Hom_{\Ch(\C)}(X, \Sigma^{-n} Y) / \sim$. \\
    We procced by consideering the complex $W = \Sigma^{-n-1} Y$ and an extension
    \[\begin{tikzcd}
        0 & W & Z & X & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    that is split degreewise. This implies that $Z_k \cong W_k \oplus X_k$, therefore by the commutativity of the diagram
    \[\begin{tikzcd}
        {W_k} & {W_k\oplus X_k} & {X_k} \\
        {W_{k-1}} & {W_{k-1} \oplus X_{k-1}} & {X_{k-1}}
        \arrow[from=1-1, to=1-2]
        \arrow["{d^W_k}", from=1-1, to=2-1]
        \arrow[from=1-2, to=1-3]
        \arrow["{d^Z_k}", from=1-2, to=2-2]
        \arrow["{d^X_k}", from=1-3, to=2-3]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
    \end{tikzcd}\]
    we know that 
    \[d_k^Z =  \begin{bmatrix}
        d_k^W & \phi_k \\
        0 & d^k_X 
        \end{bmatrix}  \]
    for some $\phi_k \colon X_k \longrightarrow W_{k-1} = \Sigma^{-n} Y$; thus we have found an element $\phi = (\phi_k)_k \in \prod_{k} \Hom(X_k, Y_{n+k})$. If we substitute the form of $d^W_k$ we just derived in the condition $d^W \circ d^W = 0$ we get that $d^Y_{k+n} \phi_k - (-1)^n \phi_k d_k^X = 0$, i.e. $\phi \in Z_k \Hom(X,Y)$.\\
    Similarly, two extensions $Z, Z'$ are equivalent iff there exists a chain isomorphism $h$ such that
    \[\begin{tikzcd}
        {W} & {Z} & {X} \\
        {W} & {Z'} & {X}
        \arrow[from=1-1, to=1-2]
        \arrow["{\id_W}", from=1-1, to=2-1]
        \arrow[from=1-2, to=1-3]
        \arrow["{h}", from=1-2, to=2-2]
        \arrow["{\id_X}", from=1-3, to=2-3]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
    \end{tikzcd}\]
    commutes, hence
    \[h_k =  \begin{bmatrix}
        1 & \psi_k \\
        0 & 1 
        \end{bmatrix}  \]
    for all $k$. If we denote as $\phi, \phi'$ the elements of $Z_k \Hom(X,Y)$ induced by $Z, Z'$, some quick matrix calculations show that $h$ is actually a morphism of chain complexes iff
    \[\phi'_k - \phi = (-1)^{n+1}d^Y_{k+n+1} h_k -h_{k-1} d^X_k.\]
    We conclude by remarking that, up to sign, this is equivalent to asking that $\phi - \phi' \in  B_k \Hom(X,Y)$.

    
\end{proof}


\begin{lemma}
    \label{lem:chain-adjoints}
    Let $\C$ be an abelian category, let $A \in ob(\C)$ and let $X,Y \in \ob(\Ch(\C))$. The following natural isomorphisms hold:
    \begin{enumerate}
		\itemsep0em 
		\item $\HomCh{\C}(D^n A, Y) = \Hom_\C(A, Y_n)$.
		\item $\HomCh{\C}(X, D^n A) = \Hom_\C(X_{n-1}, A)$.
		\item $\HomCh{\C}(S^n A, Y) = \Hom_\C(A, Z_nY)$.
		\item $\HomCh{\C}(X, S^n A) = \Hom_\C(X/B_nX, A)$.
		\item $\Ext^1_{\Ch(\C)}(D^n A, Y) = \Ext^1_\C(A, Y_n)$
		\item $\Ext^1_{\Ch(\C)}(X, D^{n+1} A) = \Ext^1_\C(X_n, A)$ 
		\item $\Ext^1_{\Ch(\C)}(S^n A, Y) = \Ext^1_\C(A, Z_nY) $
		\item $\Ext^1_{\Ch(\C)}(X, S^n A) = \Ext^1_\C(Z_nX, A)$ 
	\end{enumerate}
\end{lemma}





\begin{lemma}
    If $\C$ is a Grothendieck category, then $\Ch(\C)$ is also a Grothendieck category.
\end{lemma}


\begin{proof}
    Let $G$ be a generator of $\C$, the adjunction \[\HomCh{\C}(D^n G, X) = \Hom_\C(G, X_n)\] shows that $\{D^n G\}_{n \in \Z}$ is a generating family for $\Ch(\C)$. We conclude by remembering that colimits in $\Ch(\C)$ are taken dimensionwise.
\end{proof}




\section{Induced cotorsion pairs}
Let $\C$ be an abelian category, let $(\D, \E)$ be a cotorsion pair.

\begin{definition}[Induced cotorsion pair]
    Let $X$ be a chain complex.
    \begin{enumerate}
		\itemsep0em 
		\item $X$ is called a $\D$ \textbf{complex} if it is exact and $Z_n X \in \D$ for all $n \in \Z$.
		\item $X$ is called a $\E$ \textbf{complex} if it is exact and $Z_n X \in \E$ for all $n \in \Z$.
		\item $X$ is called a dg-$\D$ \textbf{complex} if $X_n \in \D$ for all $n \in \Z$ and $\Hom(X, B)$ is exact for all $\E$ complexes $B$.
		\item $X$ is called a dg-$\E$\textbf{complex} if $X_n \in \E$ for all $n \in \Z$ and $\Hom(A, X)$ is exact for all $\D$ complexes $A$.
	\end{enumerate}
    We denote the class of all $\D$ complexes as $\widetilde{\D}$, while the class of all dg-$\D$ complexes is denoted as $\dg{\D}$. Similarly, we denote the class of all $\E$ complexes as $\widetilde{\E}$, while the class of all dg-$\E$ complexes is denoted as $\dg{\E}$. We also use $\ex$ to denote the class of all exact complexes.
\end{definition}

It is important to remark that if $X \in \DD$ then it must be exact, so $X_n$ is an extension of $Z_nX$ and $Z_{n-1}X$; thus by lemma \ref{lem:cotorsion-trans}, $X_n \in \D$ for all $n \in \Z$. Similarly, if $X \in \EE$ then $X_n \in \E$ for all $n \in \Z$.

\begin{lemma}
    \label{lem:dg-sphere}
    Bounded below complexes (i.e. complexes $X$ where there exists $n\in \Z$ such that if $k<n$ then $X_k = 0$) with entries in $\D$ are dg-$\D$ complexes.\\
     Bounded above complexes (i.e. complexes $X$ where there exists $n\in \Z$ such that if $k>n$ then $X_k = 0$) with entries in $\E$ are dg-$\E$ complexes.
\end{lemma}

\begin{proof}
    We will prove the first statement. Let $X$ be a bounded below complex with entries in $\D$, we need to show that $\Hom(X, B)$ is exact for any $\E$ complex $B$. Given that if $B$ is an $\E$-complex then $\Sigma^k B$ is also an $\E$-complex, we only need to prove that all morphisms $f\colon X \longrightarrow B$ are null-homotopic. Let us assume without loss of generality that $X_n$ is the zero object for all negative $n$. We want to build a family of morphisms $h_k \colon X_k \longrightarrow E_{k+1}$ such that $f_k =d_{k+1}^E h_k+ h_{k-1}d^X_k$ for $k \in \Z$, as pictured below.
    \[\begin{tikzcd}
        \cdots & {X_2} & {X_1} & {X_0} & 0 & \cdots \\
        \cdots & {E_2} & {E_1} & {E_0} & {E_{-1}} & \cdots
        \arrow["{{{d^X_3}}}", from=1-1, to=1-2]
        \arrow["{{{d^X_2}}}", from=1-2, to=1-3]
        \arrow["{{{h_2}}}"{description}, from=1-2, to=2-1]
        \arrow["{{{f_2}}}", from=1-2, to=2-2]
        \arrow["{{{d^X_1}}}", from=1-3, to=1-4]
        \arrow["{{{h_1}}}"{description}, from=1-3, to=2-2]
        \arrow["{{{f_1}}}", from=1-3, to=2-3]
        \arrow["{{{d^X_0}}}", from=1-4, to=1-5]
        \arrow["{{{h_0}}}"{description}, from=1-4, to=2-3]
        \arrow["{{{f_0}}}", from=1-4, to=2-4]
        \arrow[from=1-5, to=1-6]
        \arrow["{h_{-1}}"{description}, from=1-5, to=2-4]
        \arrow[from=1-5, to=2-5]
        \arrow["{{{d^E_3}}}"', from=2-1, to=2-2]
        \arrow["{{{d^E_2}}}"', from=2-2, to=2-3]
        \arrow["{{{d^E_1}}}"', from=2-3, to=2-4]
        \arrow["{{{d^E_0}}}"', from=2-4, to=2-5]
        \arrow[from=2-5, to=2-6]
    \end{tikzcd}\]
    We start by setting $h_k = 0$ for all negative $k$, then we will prove by induction on $k$ that there is such $h_k$ for all $k\geq 0$.\\
    We want to define $h_k$ by assuming the existence of $h_{k-1}$: we start by setting $g_k := f_k - h_{k-1}d_{k}^E$, then $h_k$ is (if it exists) exactly a lift of $g_k$ through $d_{k+1}$
    \[\begin{tikzcd}
        & {X_k} \\
        {E_{k+1}} & {E_k.}
        \arrow["{h_k}"', dashed, from=1-2, to=2-1]
        \arrow["{g_k}", from=1-2, to=2-2]
        \arrow["{d^E_{k+1}}"', from=2-1, to=2-2]
    \end{tikzcd}\]
    We notice that, remembering the inductive hypothesis,
    \[d_k^E g_k = d_k^E (f_k - h_{k-1}d_{k}^E) = d_k^E f_k - f_{k-1} d_k^X  + h_{k-2}d^X_{k-1}d^X_k = 0,\]
    therefore $g_k$ factors $Z_k E =B_k E$.
    By applying the functor $\Hom_{\Ch(\C)}(X_k, {-})$ to the exact sequence\[\begin{tikzcd}
        0 & {Z_{k+1} E} & {E_{k+1}} & {B_{k} E} & {0}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    and remembering that $\Ext^1(X_k, Z_k E) = 0$, we get a surjection
    \[(d_{k+1})_* \colon \Hom_{\Ch(\C)}(X_k, E_{k+1}) \longrightarrow \Hom_{\Ch(\C)}(X_k, B_{k}),\]
    therefore a lift of $g_k$ must exist.

\end{proof}


\begin{proposition}
    If $(\D, \E)$ has enough $\D$-objects and enough $\E$-objects (in particular when it is complete), then $(\dg{\D}, \EE)$ and $(\DD, \dg{\E})$ are cotorsion pairs. 
\end{proposition}

\begin{proof}
    We will only prove the first statement; we begin by showing that $\rperp{(\dg{\D})} \subseteq \EE$. Let $X \in \rperp{(\dg{\D})}$, by applying lemmas \ref{lem:chain-adjoints} and \ref{lem:dg-sphere}, we get that, for every $D \in \D$,
    \[\Ext^1_\C(D, Z_n X) = \Ext^1_{\Ch(\C)}(S^n D, X) = 0,\]
    thus we have proven that $Z_n X \in \lperp{\D} =\E$ for all $n$. Let us now consider an epimorphism $f \colon D \longrightarrow Z_n X$ for some $n \in \Z$ and some $D \in \D$ (it must exists given that $(\D, \E)$ has enough $\D$-objects); as before $\Ext^1_{\Ch(\C)}(S^n D, X) = 0$, therefore the induced morphism $f'\colon S^n D \longrightarrow X$ must be null-homotopic and hence $f' = d^X_{n+1} h_n$ for some homotopy $\{h_i \colon (S^n D)_i \longrightarrow X_{i+1}\}_i$. This result implies that $Z_nX = \im(f') \subseteq \im(d^X_{n+1}) = B_n X$. We have concluded that $X$ must be eexact, thus $X \in \EE$. \\
    We continue by showing that $\lperp{\EE} \subseteq (\dg{\D})$. Let $X \in \lperp{\EE}$ and let $Y$ be a $\E$-complex, then $\Ext^1_{\Ch(\C)}(X,Y) = 0$ and in particular $\Ext^1_{dw}(X,Y) = 0$; so $\Hom(X,Y)$ is exact. To show that $X_n \in \D$ we can apply lemma \ref{lem:chain-adjoints} to write, for all $E\in \E$,
    \[\Ext^1_\C(X_n, E) \cong \Ext^1_{\Ch(\C)}(X,D^{n+1}E) = 0.\]
    The opposite inclusions are immediate consequences of lemma \ref{lem:deg-ext}
\end{proof}



\begin{definition}[Induced cotorsion pairs]
    Whenever the pairs $(\dg{\D}, \EE)$ and $(\DD, \dg{\E})$ are cotorsion pairs, they are said to be the \textbf{induced cotorsion pairs} (from $(\D, \E)$). We say that the induced cotorsion pairs are compatible if $\DD = \dg{\D} \cap \ex$ and $\EE = \dg{\E} \cap \ex$.
\end{definition}

We proceed by proving a rather technical proposition (due to Xiaoyan Yang and Nanqing Ding) that we will need later to prove the completeness (in under certain hypothesis) of the induced cotorsion pair; to do so we need some preliminary algebraic results.


\begin{lemma}
    \label{lem:inv-limit-chain}
    Let us assume $\C$ to be complete and let's consider an inverse system in $\Ch(\C)$
    \[\begin{tikzcd}
        \cdots & {P_3} & {P_2} & {P_1} & {P_0.}
        \arrow[from=1-1, to=1-2]
        \arrow["{\mu_3}", from=1-2, to=1-3]
        \arrow["{\mu_2}", from=1-3, to=1-4]
        \arrow["{\mu_1}", from=1-4, to=1-5]
    \end{tikzcd}\]
    We can define a morphism $\nu \colon \prod_{i \in \N} P_i \longrightarrow \prod_{i \in \N} P_i$ induced by 
    \[\begin{tikzcd}
        {\prod_{i \in \N} P_i} & {\prod_{i \in \N} P_i} \\
        {P_{n+1}} & {P_n.}
        \arrow["\nu", from=1-1, to=1-2]
        \arrow["{\pi_{n+1}}"', from=1-1, to=2-1]
        \arrow["{\pi_n}", from=1-2, to=2-2]
        \arrow["{\mu_{n+1}}", from=2-1, to=2-2]
    \end{tikzcd}\]
    If we assume each $\mu_i$ to be epic and dimensionwise split, the following sequence is split exact in each degree.
    \[\begin{tikzcd}
    0 
    \arrow[r] 
    & \varprojlim P_n 
    \arrow[r, "\lambda"] 
    & \displaystyle\prod_{n \geq 0} P_n 
    \arrow[r, "1-\nu"] 
    & \displaystyle\prod_{n \geq 0} P_n 
    \arrow[r] 
    & 0
  \end{tikzcd}\]
\end{lemma}

\begin{proof}
    See \cite{MurfetDerived}, definition 29, lemma 67 and proposition 85.
\end{proof}

\begin{proposition}
    Let us assume that $\C$ is complete and cocomplete and let $X \in \Ch(\C)$.
    \begin{enumerate}
        \itemsep0em 
        \item If $(\D, \E)$ has enough $\E$-objects, then there exists an exact sequence in $\Ch(\C)$
            \[\begin{tikzcd}
                0 & E & P & X & {0,}
                \arrow[from=1-1, to=1-2]
                \arrow[from=1-2, to=1-3]
                \arrow[from=1-3, to=1-4]
                \arrow[from=1-4, to=1-5]
            \end{tikzcd}\]
            where $E \in \dg{\E}$ and $P \in \ex$.
        \item If $(\D, \E)$ has enough $\D$-objects, then there exists an exact sequence in $\Ch(\C)$
            \[\begin{tikzcd}
                0 & X & P & D & {0,}
                \arrow[from=1-1, to=1-2]
                \arrow[from=1-2, to=1-3]
                \arrow[from=1-3, to=1-4]
                \arrow[from=1-4, to=1-5]
            \end{tikzcd}\]
            where $D \in \dg{\D}$ and $P \in \ex$.
    \end{enumerate}
\end{proposition}




\begin{proof}
    We will only prove the first statement, given that the other is dual. Let us assume that $(\D, \E)$ has enough $\E$-objects and let $X \in \Ch(\C)$. We set $C_n := \coker(d_{n+1}^X)$ and let $\rho_n := X_{n} \longrightarrow C_n$ be the natural epimorphism. We will now define a new complex $Y^0$ by performing the so-called "killing boundaries construction": let us choose an integer $n_0 \in \Z$ and a monomorphism \[\iota \colon C_{n_0} \longrightarrow E_{n_0-1}\] for $E_{n_0-1} \in \E$ (there are enough $\E$-objects), so that the objects of the complex are $Y^0_{n_0-1} := X_{n-1} \oplus E_{n_0-1}$ and $Y^0_k := X_k$ ($k \neq n_0$), while the boundary  is given by 
    \[\begin{tikzcd}
        \cdots & {X_{n+1}} & {X_n} & {X_{n-1} \oplus E_{n_0-1}} & {X_{n-2}} & \cdots
        \arrow[from=1-1, to=1-2]
        \arrow["{d^X_{n+1}}", from=1-2, to=1-3]
        \arrow["{(d^X_{n}, \iota \rho_n)}", from=1-3, to=1-4]
        \arrow["{d^X_{n-1}}", from=1-4, to=1-5]
        \arrow[from=1-5, to=1-6]
    \end{tikzcd}\]
    It is clear that we get a morphism $Y^0 \longrightarrow X$ which is a degreewise split epi and whose kernel is $E^0 := S^{n_0-1}E_{n_0 -1} \in \dg{\E}$. It is easy to see that  $Y^0$ is exact at $n_0$ and that the induced morphism $B_kY^0 \longrightarrow B_k X$ is an isomorphism for $k \neq n_0$. Given a sequence $(n_0, n_1, n_2, \dots)$ such that every integer appears in it infinitely many times we can repeat the killing boundaries construction to obtain a sequence of complexes $Y^l$ and degreewise split epis $Y^{l+1} \longrightarrow Y^l$ such that $Y^l$ is exact at position $n_l$, $B_kY^{l+1} \longrightarrow B_k Y^l$ is an isomorphism for $k \neq l+1$ and \[\ker(Y^{l+1} \longrightarrow Y^l) = S^{n_l-1}E_{n_l -1} \] is in $\dg{\E}$\\
    We continue by defining, just like we did for $l=0$, the complex
    \[E^l := \ker(Y^l \longrightarrow Y^{l-1} \longrightarrow \dots \longrightarrow Y^0 \longrightarrow X)\]
    and remarking that the induced morphism $E^{l+1} \longrightarrow E^l$ must be degreewise split epis ($\ker$ is a functor, so it sends split epis in the arrow category to split epis); it is also important to remark that $E^{l+1}$ is an extension of $S^{n_l-1}E_{n_l -1}$ and $E^l$, therefore by induction it must also be in $\dg{\E}$. Furthermore for all $l\in \N$ we obtain the following degreewise split exact sequences (composition of split epis is a split epi):
    \[\begin{tikzcd}
        0 & {E^l} & {Y^l} & X & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    Glueing the sequences associated to each degree yields 
    \[\begin{tikzcd}
        0 & {\displaystyle\prod_{l= 0}^{\infty} E^l_k} & {\displaystyle\prod_{l= 0}^{\infty} Y^l_k} & {\displaystyle\prod_{l= 0}^{\infty} X_k} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    which is split exact for all $k \in \Z$, therefore
    \[\begin{tikzcd}
        0 & {\displaystyle\prod_{l= 0}^{\infty} E^l} & {\displaystyle\prod_{l= 0}^{\infty} Y^l} & {\displaystyle\prod_{l= 0}^{\infty} X} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    is exact. We continue by setting $E := \varprojlim E^l$ and $P:= \varprojlim Y^l$ (and remarking that, trivially, $X = \varprojlim X$) and applying lemma \ref{lem:inv-limit-chain} to obtain a commutative diagram with exact columns and where the last two rows are also exact;
    \[\begin{tikzcd}
        & 0 \arrow[d] & 0 \arrow[d] & 0 \arrow[d] & \\
        0 \arrow[r] 
        & E \arrow[r] \arrow[d] 
        & P \arrow[r] \arrow[d] 
        & X \arrow[r] \arrow[d] 
        & 0 \\
        0 \arrow[r] 
        & \displaystyle\prod_{l = 0} ^{\infty} E^l \arrow[r] \arrow[d, "1-\nu"] 
        & \displaystyle\prod_{l = 0} ^{\infty} Y^l \arrow[r] \arrow[d, "1-\nu"] 
        & \displaystyle\prod_{l = 0} ^{\infty} X \arrow[r] \arrow[d, "1-\nu"] 
        & 0 \\
        0 \arrow[r] 
        & \displaystyle\prod_{l = 0} ^{\infty} E^l \arrow[r] \arrow[d] 
        & \displaystyle\prod_{l = 0} ^{\infty} Y^l \arrow[r] \arrow[d] 
        & \displaystyle\prod_{l = 0} ^{\infty} X \arrow[r] \arrow[d] 
        & 0 \\
        & 0 & 0 & 0 &
    \end{tikzcd}\]
    from the nine lemma follows the exactness of the sequence
    \[\begin{tikzcd}
        0 & E & P & X & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    To prove that this sequence is the one we are looking for, it remains to prove that $E \in \dg{\E}$ and $P \in \ex$, we begin by investigating $E$. The first step is proving that $E_k \in \E$ for all $k$, but if we set $K^l := \ker(E^{l+1} \longrightarrow E^l)$, the following diagram of kernels 
    \[\begin{tikzcd}
        0 & {K^l} & {S^{n_l-1}E_{n_l-1}} & 0 & \\
        0 & {E^{l+1}} & {Y^{l+1}} & X & 0 \\
        0 & {E^l} & {Y^l} & X & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-2, to=2-2]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-3, to=2-3]
        \arrow[from=1-4, to=2-4]
        \arrow[from=2-1, to=2-2]
        \arrow[from=2-2, to=2-3]
        \arrow[from=2-2, to=3-2]
        \arrow[from=2-3, to=2-4]
        \arrow[from=2-3, to=3-3]
        \arrow[from=2-4, to=2-5]
        \arrow[from=2-4, to=3-4]
        \arrow[from=3-1, to=3-2]
        \arrow[from=3-2, to=3-3]
        \arrow[from=3-3, to=3-4]
        \arrow[from=3-4, to=3-5]
    \end{tikzcd}\]
    shows that $K^l \cong S^{n_l-1}E_{n_l-1} \in \dg{\E}$. Given that the morphism in the projective system used to compute $E$ are degreewise split epis, we can compute $E_k$ degreewise to find
    \[ E_k = \left( \varprojlim E^l \right)_k = E^0_k \oplus \prod_{l=0}^{\infty} K^l_k,\]
    thus $E_k\in \E$, given that $\E$ is part of a cotorsion pair and therefore closed under products (see lemma \ref{lem:cotorsion-prod}). We can apply the $\Hom(D, {-})$ functor (for a certain $D \in \D$) to the short exact sequence 
    \[\begin{tikzcd}
        0 & {K^l} & {E^{l+1}} & {E^l} & 0
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    to obtain 
    \[\begin{tikzcd}[sep=small]
        \cdots & {\Hom_\C(D, E^{l+1})} & {\Hom_\C(D, E^l)} & {\Ext^1_\C(D, K^l)} & {\cdots.}
	    \arrow[from=1-1, to=1-2]
	    \arrow[from=1-2, to=1-3]
	    \arrow[from=1-3, to=1-4]
	    \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    Given that $K^l \in \dg{\E}$ and $D \in \D$, we know that ${\Ext^1_\C(D, K^l)} = 0$, therefore the morphism 
    \[{\Hom_\C(D, E^{l+1})} \longrightarrow {\Hom_\C(D, E^l)}\]
    must be epic for all $l$. Applying lemma \ref{lem:inv-limit-group}, we obtain the following exact sequence:
    \[\begin{tikzcd}[sep=tiny]
        0 & {\varprojlim \Hom_\C(D,E^l)} & {\displaystyle \prod_{l=0}^{\infty}\Hom_\C(D,E^l)} & {\displaystyle \prod_{l=0}^{\infty}\Hom_\C(D,E^l)} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow[from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    hence, given that $\prod_{l=0}^{\infty}\Hom_\C(D,E^l)$ is exact ($E^l \in \dg{\E}$ and the product of exact complexes of abelian groups is exact), $\varprojlim \Hom_\C(D,E^l) \cong \Hom_\C(D,E)$ is exact ($\Hom(D, {-})$ is a continuous functor).\\
    All that remains is showing that $P \in \ex$, we will show that it is exact at each homological position $k \in \Z$. Let us choose a cofinal subsequence $(n_{m_0}, n_{m_1}, \dots)$ of $(n_0, n_1, \dots)$ such that $n_{m_t} = k$ for all $t$; such a subsequence must exist given that each integer appears infinitely many times in the original sequence, furthermore the cofinality condition implies that we can compute the limit using only complexes that are exact in position $k$:
    \[P = \varprojlim_{i \in \N} Y^i = \varprojlim_{i \in \N} Y^{m_i}.\]
    Applying lemma \ref{lem:inv-limit-chain} allows us to write the following
    \[\begin{tikzcd}
        0 & P & {\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}} & {\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}} & {0,}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow["1-\nu", from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    which is split exact in each degree and yields the following exact sequence:
    \[\begin{tikzcd}
        { 0} & {H_k(P)} & {H_k\left({\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}}\right)} & {H_k\left({\displaystyle\prod_{i = 0} ^{\infty} Y^{m_i}}\right)} & {0.}
        \arrow[from=1-1, to=1-2]
        \arrow[from=1-2, to=1-3]
        \arrow["{H_k(1- \nu)}", from=1-3, to=1-4]
        \arrow[from=1-4, to=1-5]
    \end{tikzcd}\]
    If we denote as $\mu_j$ the morphism $Y^{m_j} \longrightarrow Y^{m_{j-1}}$, from the definition of $\nu$ follows immediately that \[\pi_{m_j} (1-\nu) = \pi_{m_j} - \mu_{j+1} \pi_{{m_{j+1}}}\] for all $j$; from the exactness of $Y^l$ and the fact that for abelian groups products are exact, by denoting as $\mu_{t,i}$ and $\pi_{t,i}$ the degree $i$ component of $\mu_t$ and $\pi_t$, we deduce  
    \begin{align*}
        \mu_{j+1, k} \pi_{m_{j+1}, k} \left(Z_k \prod_{i=0}^{\infty} Y^{m_i}  \right) &=  \mu_{j+1, k} \pi_{m_{j+1}, k}\left( \prod_{i=0}^{\infty} Z_k  Y^{m_i}  \right) = \\
        &= \mu_{j+1, k} \left( Z_k Y^{m_{j+1}} \right) = \\
        &= \mu_{j+1, k}  \left(B_k Y^{m_{j+1}} \right) = \\
        &= \mu_{j+1, k} d^{Y^{m_{j+1}}}_{k+1} \left(Y^{m_{j+1}}_{k+1} \right) = \\
        &=  d^{Y^{m_{j}}}_{k+1} \mu_{j+1, k+1} \left(Y^{m_{j+1}}_{k+1} \right) \subseteq B_k Y^{m_j}.
    \end{align*}
    We conclude that $H_k (\mu_{j+1} \pi_{{m_{j+1}}})$ is trivial, therefore $H_k(\pi_{m_j}(1-\nu)) = H_k(\pi_j)$ for all $j$, thus $1-\nu$ is a quasi-isomorphism and hence $H_k(P) = 0$. Given that this holds for all $k$, $P$ must be exact.
    



    
\end{proof}
